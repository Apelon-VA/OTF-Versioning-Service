113     public ConceptChronicleDdo(TerminologySnapshotDI ss, ConceptChronicleBI c, VersionPolicy versionPolicy,
114             RefexPolicy refexPolicy, RelationshipPolicy relationshipPolicy)
115             throws IOException, ContradictionException {
116         this.versionPolicy = versionPolicy;
117         this.refexPolicy = refexPolicy;
118         this.relationshipPolicy = relationshipPolicy;
119         this.viewCoordinateUuid = ss.getViewCoordinate().getVcUuid();
120         if (ss.getConceptForNid(c.getNid()).getPreferredDescription() != null) {
121             this.conceptReference = new ComponentReference(c.getPrimordialUuid(), c.getNid(),
122                     ss.getConceptForNid(c.getNid()).getPreferredDescription().getText());
123         } else {
124             this.conceptReference = new ComponentReference(c.getNid());
125 
126         }
127         this.conceptAttributes = new ConceptAttributesChronicleDdo(ss, this, c.getConceptAttributes());
128         this.primordialUuid = conceptAttributes.getPrimordialComponentUuid();
129 
130         switch (relationshipPolicy) {
131             case DESTINATION_RELATIONSHIPS:
132                 _destinationRelationships =
133                         FXCollections.observableArrayList(new ArrayList<RelationshipChronicleDdo>(0));
134                 addDestinationRelationships(c, ss);
135 
136                 break;
137 
138             case ORIGINATING_RELATIONSHIPS:
139                 _originRelationships = FXCollections.observableArrayList(new ArrayList<RelationshipChronicleDdo>(0));
140                 addOriginRelationships(c, ss);
141 
142                 break;
143 
144             case ORIGINATING_AND_DESTINATION_RELATIONSHIPS:
145                 addOriginRelationships(c, ss);
146                 addDestinationRelationships(c, ss);
147 
148                 break;
149 
150             case ORIGINATING_AND_DESTINATION_TAXONOMY_RELATIONSHIPS:
151                 addOriginTaxonomyRelationships(c, ss);
152                 addDestinationTaxonomyRelationships(c, ss);
153 
154                 break;
155 
156             default:
157                 throw new UnsupportedOperationException("Can't handle: " + relationshipPolicy);
158         }
159 
160         _descriptions =
161                 FXCollections.observableArrayList(new ArrayList<DescriptionChronicleDdo>(c.getDescriptions().size()));
162 
163         for (DescriptionChronicleBI desc : c.getDescriptions()) {
164             DescriptionChronicleDdo dc = new DescriptionChronicleDdo(ss, this, desc);
165 
166             if (!dc.getVersions().isEmpty()) {
167                 _descriptions.add(dc);
168             }
169         }
170 
171         _media = FXCollections.observableArrayList(new ArrayList<MediaChronicleDdo>(c.getMedia().size()));
172 
173         for (MediaChronicleBI mediaChronicle : c.getMedia()) {
174             MediaChronicleDdo tkMedia = new MediaChronicleDdo(ss, this, mediaChronicle);
175 
176             if (!tkMedia.getVersions().isEmpty()) {
177                 _media.add(tkMedia);
178             }
179         }
180 
181         if (((refexPolicy == RefexPolicy.ANNOTATION_MEMBERS_AND_REFSET_MEMBERS)
182                 || (refexPolicy == RefexPolicy.REFEX_MEMBERS_AND_REFSET_MEMBERS)) && !c.isAnnotationStyleRefex()) {
183             Collection<? extends RefexChronicleBI> members = c.getRefsetMembers();
184 
185             if (members != null) {
186                 _refsetMembers = FXCollections.observableArrayList(new ArrayList<RefexChronicleDdo<?, ?>>(members.size()));
187 
188                 for (RefexChronicleBI m : members) {
189                     RefexChronicleDdo<?, ?> member = convertRefex(ss, m);
190 
191                     if ((member != null) && !member.getVersions().isEmpty()) {
192                         _refsetMembers.add(member);
193                     } else {
194                         throw new IOException("Could not convert refset member: " + m + "\nfrom refset: " + c);
195                     }
196                 }
197             }
198         }
199     }

219     protected final void addDestinationTaxonomyRelationships(ConceptChronicleBI c, TerminologySnapshotDI ss)
220             throws ContradictionException, IOException {
221         Collection<? extends RelationshipChronicleBI> relsIncoming = c.getRelationshipsIncoming();
222 
223         _destinationRelationships =
224                 FXCollections.observableArrayList(new ArrayList<RelationshipChronicleDdo>(relsIncoming.size()));
225 
226         int isaNid = ss.getViewCoordinate().getIsaNid();
227 
228         NEXT_REL:
229         for (RelationshipChronicleBI rel : relsIncoming) {
230             RelationshipVersionBI relVersion = rel.getPrimordialVersion();
231 
232             switch (ss.getViewCoordinate().getRelationshipAssertionType()) {
233                 case INFERRED:
234                     if (!relVersion.isInferred()) {
235                         continue NEXT_REL;
236                     }
237 
238                     break;
239 
240                 case STATED:
241                     if (!relVersion.isStated()) {
242                         continue NEXT_REL;
243                     }
244 
245                     break;
246 
247                 case INFERRED_THEN_STATED:
248                     if (!relVersion.isInferred()) {
249                         continue NEXT_REL;
250                     }
251 
252                     break;
253             }
254 
255             boolean foundType = false;
256 
257             for (RelationshipVersionBI rv : rel.getVersions(ss.getViewCoordinate())) {
258                 if (isaNid == rv.getTypeNid()) {
259                     foundType = true;
260 
261                     break;
262                 }
263             }
264 
265             if (!foundType) {
266                 continue NEXT_REL;
267             }
268 
269             RelationshipChronicleDdo fxc = new RelationshipChronicleDdo(ss, this, rel);
270 
271             _destinationRelationships.add(fxc);
272         }
273 
274         if (_destinationRelationships.isEmpty()
275                 && (ss.getViewCoordinate().getRelationshipAssertionType()
276                 == RelAssertionType.INFERRED_THEN_STATED)) {
277             for (RelationshipChronicleBI rel : relsIncoming) {
278                 RelationshipVersionBI relVersion = rel.getPrimordialVersion();
279 
280                 if (relVersion.isStated()) {
281                     boolean foundType = false;
282 
283                     for (RelationshipVersionBI rv : rel.getVersions(ss.getViewCoordinate())) {
284                         if (isaNid != rv.getTypeNid()) {
285                             foundType = true;
286 
287                             break;
288                         }
289                     }
290 
291                     if (foundType) {
292                         RelationshipChronicleDdo fxc = new RelationshipChronicleDdo(ss, this, rel);
293 
294                         _destinationRelationships.add(fxc);
295                     }
296                 }
297             }
298         }
299     }

315     protected final void addOriginTaxonomyRelationships(ConceptChronicleBI c, TerminologySnapshotDI ss)
316             throws ContradictionException, IOException {
317         _originRelationships = FXCollections.observableArrayList(
318                 new ArrayList<RelationshipChronicleDdo>(c.getRelationshipsOutgoing().size()));
319 
320         int isaNid = ss.getViewCoordinate().getIsaNid();
321 
322         for (RelationshipChronicleBI rel : c.getRelationshipsOutgoing()) {
323             RelationshipChronicleDdo fxc = new RelationshipChronicleDdo(ss, this, rel);
324             ArrayList<RelationshipVersionDdo> toRemove = new ArrayList<>();
325 
326             for (RelationshipVersionDdo fxv : fxc.getVersions()) {
327                 if (isaNid != fxv.getTypeReference().getNid()) {
328                     toRemove.add(fxv);
329 
330                     break;
331                 }
332 
333                 switch (ss.getViewCoordinate().getRelationshipAssertionType()) {
334                     case INFERRED:
335                         if (!(fxv.getAuthorReference().getNid() == ss.getViewCoordinate().getClassifierNid())) {
336                             toRemove.add(fxv);
337                         }
338 
339                         break;
340 
341                     case STATED:
342                         if (fxv.getAuthorReference().getNid() == ss.getViewCoordinate().getClassifierNid()) {
343                             toRemove.add(fxv);
344                         }
345 
346                         break;
347                 }
348             }
349 
350             fxc.getVersions().removeAll(toRemove);
351 
352             if (!fxc.getVersions().isEmpty()) {
353                 _originRelationships.add(fxc);
354             }
355         }
356     }
