48     @GET
49     @Path("{id}/{vcUuid}")
50     @Produces("application/bdb")
51     public StreamingOutput getConceptByteArray(@PathParam("id") String id, @PathParam("vcUuid") String vcUuid)
52             throws IOException {
53        final int cnid;
54  
55        if (id.length() == 36) {
56           UUID uuid = UUID.fromString(id);
57  
58           cnid = BdbSingleton.get().getNidForUuids(uuid);
59        } else {
60           cnid = Integer.parseInt(id);
61        }
62  
63        final ConceptDataFetcherI fetcher = BdbSingleton.get().getConceptDataFetcher(cnid);
64  
65        return new StreamingOutput() {
66           @Override
67           public void write(OutputStream output) throws IOException, WebApplicationException {
68              DataOutputStream dos = new DataOutputStream(output);
69  
70              dos.writeInt(cnid);
71  
72              byte[] robs = fetcher.getReadOnlyBytes();
73  
74              dos.writeInt(robs.length);
75              dos.write(robs);
76  
77              byte[] rwbs = fetcher.getReadWriteBytes();
78  
79              dos.writeInt(rwbs.length);
80              dos.write(rwbs);
81           }
82        };
83     }
