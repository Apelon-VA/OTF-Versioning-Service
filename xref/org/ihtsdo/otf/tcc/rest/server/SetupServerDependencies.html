40  public class SetupServerDependencies {
41  
42      ServletContext context;
43  
44      public SetupServerDependencies(ServletContext context) {
45          this.context = context;
46      }
47      
48      public boolean execute() throws IOException {
49          return execute(null);
50      }
51  
52      public boolean execute(String args[]) throws IOException {
53          String appHome = System.getenv("CATALINA_HOME") + "/temp/bdb";
54  
55          context.log("App home: " + appHome);
56  
57          File app = new File(appHome);
58  
59          if (!app.exists()) {
60              app.mkdirs();
61          }
62  
63          context.log("The default user settings file " + MavenCli.DEFAULT_USER_SETTINGS_FILE.getAbsolutePath());
64          System.setProperty("org.slf4j.simpleLogger.showLogName", "false");
65          System.setProperty("org.slf4j.simpleLogger.showThreadName", "false");
66          System.setProperty("org.slf4j.simpleLogger.levelInBrackets", "true");
67          System.setProperty("org.ihtsdo.otf.tcc.datastore.working-dir", appHome + "/target");
68          System.setProperty("org.ihtsdo.otf.tcc.datastore.bdb-location", appHome + "/target/berkeley-db");
69  
70          MavenCli cli = new MavenCli();
71  
72          // Write to the pom.xml
73          File pomDir = new File(appHome);
74  
75          if (!pomDir.exists()) {
76              pomDir.mkdirs();
77          }
78  
79          String username = System.getProperty("org.ihtsdo.otf.tcc.repository.username");
80          if (username == null || username.isEmpty()) {
81              context.log("WARNING: Username is null. Please set "
82                      + "'org.ihtsdo.otf.tcc.repository.username'  to a proper "
83                      + "value in the CATALINA_OPTS environmental variable");
84          }
85          String password = System.getProperty("org.ihtsdo.otf.tcc.repository.password");
86          if (password == null || password.isEmpty()) {
87              context.log("WARNING: Password is null. Please set "
88                      + "'org.ihtsdo.otf.tcc.repository.password' to a proper "
89                      + "value in the CATALINA_OPTS environmental variable");
90          }
91          String pomResource = "/WEB-INF/classes/org/ihtsdo/otf/serversetup/pom.xml";
92          String settingsResource = "/WEB-INF/classes/org/ihtsdo/otf/serversetup/settings.xml";
93  
94          context.log("pom: " + context.getResource(pomResource));
95  
96          File pomFile = new File(pomDir + "/pom.xml");
97          if (pomFile.exists()) {
98              context.log("Pom file exists. Now deleting.");
99              pomFile.delete();
100         }
101         BufferedReader pomReader =
102                 new BufferedReader(new InputStreamReader(context.getResourceAsStream(pomResource), "UTF-8"));
103         BufferedWriter pomWriter = new BufferedWriter(new OutputStreamWriter(new FileOutputStream(pomFile), "UTF-8"));
104 
105         try {
106             String settingsLine;
107 
108             while ((settingsLine = pomReader.readLine()) != null) {
109                 pomWriter.write(settingsLine + "\n");
110             }
111         } finally {
112             pomReader.close();
113             pomWriter.close();
114         }
115 
116         // Write the settings.xml file
117 
118         context.log("settings: " + context.getResource(settingsResource));
119         File settingsFile = new File(appHome, "settings.xml");
120         if (settingsFile.exists()) {
121             context.log("Settings file exists. Now deleting.");
122             settingsFile.delete();
123         }
124         context.log("settings path: " + settingsFile.getAbsolutePath());
125 
126         BufferedReader settingsReader =
127                 new BufferedReader(new InputStreamReader(context.getResourceAsStream(settingsResource), "UTF-8"));
128         BufferedWriter settingsWriter = new BufferedWriter(
129                 new OutputStreamWriter(
130                 new FileOutputStream(settingsFile), "UTF-8"));
131         
132         try {
133             String s;
134 
135             while ((s = settingsReader.readLine()) != null) {
136                 s = s.replace("<username>user</username>", "<username>" + username + "</username>");
137                 s = s.replace("<password>password</password>", "<password>" + password + "</password>");
138                 s = s.replace("<localRepository>mvn-repo</localRepository>", "<localRepository>" + appHome + "/mvn-repo</localRepository>");
139                 settingsWriter.write(s + "\n");
140             }
141         } finally{
142             settingsReader.close();
143             settingsWriter.close();
144         }
145 
146         if ((args == null) || (args.length == 0)) {
147             args = new String[]{"-e", 
148                 "-settings", settingsFile.getAbsolutePath(),
149                 "-U",
150                 "clean", "install"};
151         }
152 
153         OutputStream stringStream = new ContextLoggerStream();
154         PrintStream mavenOutputStream = new PrintStream(stringStream);
155         
156         
157         int result = cli.doMain(args, pomDir.getAbsolutePath(), mavenOutputStream,
158                 mavenOutputStream);
159 
160         context.log(stringStream.toString());
161 
162         if (result == 0) {
163             context.log("Embedded maven build succeeded: " + result);
164         } else {
165             context.log("Embedded maven build failed: " + result);
166             return false;
167         }
168         return true;
169     }
170     
171     private class ContextLoggerStream extends OutputStream {
172 
173         ByteArrayOutputStream bytes = new ByteArrayOutputStream();
174         
175         @Override
176         public void write(int b) throws IOException {
177             if (b == '\n' || b == '\r') {
178                 context.log(bytes.toString());
179                 bytes.reset();
180             } else {
181                 bytes.write(b);
182             }
183         }
184         
185     }
