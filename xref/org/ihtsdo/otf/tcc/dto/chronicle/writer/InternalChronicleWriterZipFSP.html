34  public class InternalChronicleWriterZipFSP extends FileChronicleZipFSP implements InternalChronicleWriterBI {
35  
36      public InternalChronicleWriterZipFSP(Path chroniclePath) throws IOException {
37          super(chroniclePath, AccessType.CREATE);
38      }
39  
40      @Override
41      public void write(TtkConceptChronicle ttkConceptChronicle, long time) throws IOException {
42  
43          ConceptChronicleBI chronicleToWrite = ChronicleConverter.convert(ttkConceptChronicle);
44          write(chronicleToWrite, time);
45      }
46  
47      @Override
48      public void write(ConceptChronicleBI chronicleToWrite, long time) throws IOException {
49          Path chroniclePath =
50                  chronicleFileSystem.getPath("c|"
51                  + Integer.toString(chronicleToWrite.getNid()) + "|" + Long.toString(time));
52          try (OutputStream out = getZipFSProvider().newOutputStream(chroniclePath, StandardOpenOption.CREATE_NEW)) {
53              chronicleToWrite.writeExternal(new DataOutputStream(out));
54          }
55          Files.setAttribute(chroniclePath, "creationTime",
56                                         FileTime.fromMillis(time));
57          Files.setAttribute(chroniclePath, "lastModifiedTime",
58                                         FileTime.fromMillis(time));
59      }
60  }
