67      public ExternalChronicleWriterZipFSP(Path chroniclePath) throws IOException {
68          super(chroniclePath, AccessType.CREATE);
69      }
70  
71      @Override
72      public void write(TtkConceptChronicle ttkConceptChronicle, long time) throws IOException {
73          Path chroniclePath =
74                  chronicleFileSystem.getPath("c|"
75                  + ttkConceptChronicle.getPrimordialUuid().toString() + "|" + Long.toString(time));
76          
77          
78          try (OutputStream out = getZipFSProvider().newOutputStream(chroniclePath, StandardOpenOption.CREATE_NEW)) {
79              ttkConceptChronicle.writeExternal(new DataOutputStream(out));
80          }
81          Files.setAttribute(chroniclePath, "creationTime",
82                                         FileTime.fromMillis(time));
83          Files.setAttribute(chroniclePath, "lastModifiedTime",
84                                         FileTime.fromMillis(time));
85      }
86  
87      @Override
88      public void write(ConceptChronicleBI chronicleToWrite, long time) throws IOException {
89          write(new TtkConceptChronicle(chronicleToWrite), time);
90      }
