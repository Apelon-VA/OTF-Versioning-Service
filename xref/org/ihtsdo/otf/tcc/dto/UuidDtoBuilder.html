98      * Method description
99      *
100     *
101     * @param blueprint
102     *
103     * @return
104     *
105     * @throws ContradictionException
106     * @throws IOException
107     * @throws InvalidCAB
108     */
109    public TtkConceptChronicle construct(ConceptCB blueprint)
110            throws IOException, InvalidCAB, ContradictionException {
111       TtkConceptChronicle newC = new TtkConceptChronicle();
112 
113       newC.setAnnotationStyleRefex(blueprint.isAnnotationRefexExtensionIdentity());
114       newC.setPrimordialUuid(blueprint.getComponentUuid());
115       construct(blueprint.getConceptAttributeAB(), newC);
116 
117       List<DescriptionCAB>  fsnBps   = blueprint.getFullySpecifiedNameCABs();
118       List<DescriptionCAB>  prefBps  = blueprint.getPreferredNameCABs();
119       List<DescriptionCAB>  descBps  = blueprint.getDescriptionCABs();
120       List<RelationshipCAB>   relBps   = blueprint.getRelationshipCABs();
121       List<MediaCAB> mediaBps = blueprint.getMediaCABs();
122 
123       for (DescriptionCAB fsnBp : fsnBps) {
124          this.construct(fsnBp, newC);
125       }
126 
127       for (DescriptionCAB prefBp : prefBps) {
128          this.construct(prefBp, newC);
129       }
130 
131       for (DescriptionCAB descBp : descBps) {
132          if (fsnBps.contains(descBp) || prefBps.contains(descBp)) {
133             continue;
134          } else {
135             this.construct(descBp, newC);
136          }
137       }
138 
139       for (RelationshipCAB relBp : relBps) {
140          this.construct(relBp, newC);
141       }
142 
143       for (MediaCAB mediaBp : mediaBps) {
144          construct(mediaBp, newC);
145       }
146 
147       return newC;
148    }
