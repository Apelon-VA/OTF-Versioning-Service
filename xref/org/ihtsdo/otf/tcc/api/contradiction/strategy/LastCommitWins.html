28  /**
29   * "Last commit wins" implementation of a conflict resolution strategy.
30   * Specifically this resolution
31   * strategy will resolve conflict by choosing the part or tuple of an entity
32   * with the latest
33   * commit time regardless of path.
34   * 
35   * "In conflict" is rare for this strategy. As the latest part of any given
36   * entity
37   * is chosen, and the differences between latest state on different paths is of
38   * little importance the only
39   * way to achieve a conflict is for two parts to have the same effective time.
40   * This is rare, but
41   * if it occurs will be considered a conflict.
42   * 
43   * When in conflict the "resolve" methods will arbitrarily return one of the
44   * parts with the most
45   * recent commit time.
46   * 
47   * @author Dion
48   */
49  @XmlRootElement(name = "last-commit-wins-strategy")

69      private <T extends ComponentVersionBI> Collection<List<T>> getSortedVersionsCopy(List<T> originalVersions) {
70          Map<Integer, List<T>> map = new HashMap<>();
71  
72          for (T v : originalVersions) {
73              List<T> versions;
74              if (map.containsKey(v.getNid())) {
75                  versions = map.get(v.getNid());
76              } else {
77                  versions = new ArrayList<>();
78              }
79              versions.add(v);
80              map.put(v.getNid(), versions);
81          }
82  
83          for (List<T> list : map.values()) {
84              Collections.sort(list, new VersionDateOrderSortComparator(true));
85          }
86  
87          return map.values();
88      }

90      private <T extends ComponentVersionBI> List<T> getLatestVersions(List<T> versions) {
91          Collection<List<T>> sortedVersions = getSortedVersionsCopy(versions);
92  
93          List<T> returnList = new ArrayList<>();
94  
95          for (List<T> v : sortedVersions) {
96              Iterator<T> iterator = v.iterator();
97              T first = iterator.next();
98              returnList.add(first);
99              T version;
100             while (iterator.hasNext() && (version = iterator.next()).getTime() == first.getTime()) {
101                 returnList.add(version);
102             }
103         }
104 
105         return returnList;
106     }

159     private <T extends ComponentVersionBI> List<T> getSortedPartsCopy(List<T> versions) {
160         List<T> copy = new ArrayList<>(versions);
161         Collections.sort(copy, new PartDateOrderSortComparator(true));
162 
163         return copy;
164     }

166     @Override
167     public <T extends ComponentVersionBI> List<T> resolveVersions(List<T> tuples) {
168         if (tuples == null || tuples.isEmpty()) {
169             return tuples;
170         }
171 
172         return getLatestVersions(tuples);
173     }
174     
175     @Override
176     public <T extends ComponentVersionBI> List<T> resolveVersions(T part1, T part2) {
177         ArrayList<T> values = new ArrayList<>();
178         if (part1.getTime() > part2.getTime()) {
179             values.add(part1);
180         } else if (part1.getTime() < part2.getTime()) {
181             values.add(part2);
182         } else {
183             values.add(part1);
184             values.add(part2);
185         }
186         return values;
187     }
