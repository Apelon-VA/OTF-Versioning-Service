17  package org.ihtsdo.otf.tcc.api.io;

19  import java.io.BufferedInputStream;
20  import java.io.File;
21  import java.io.FileInputStream;
22  import java.io.IOException;
23  import java.util.Date;
24  import java.util.Enumeration;
25  import java.util.jar.JarEntry;
26  import java.util.jar.JarFile;
27  import java.util.jar.JarInputStream;
28  
29  public class JarExtractor {
30  
31      private enum READ_TYPE {
32          JAR_FILE, JAR_INPUT_STREAM
33      };
34  
35      private static READ_TYPE readType = READ_TYPE.JAR_INPUT_STREAM;
36  
37      /**
38       * Note the following about opening large jar files:
39       * <p>
40       * http://confluence.atlassian.com/display/DOC/java.util.zip.ZipFile.open+
41       * causes+OutOfMemoryError+for+large+zip+files
42       * <p>
43       * http://bugs.sun.com/bugdatabase/view_bug.do;:YfiG?bug_id=4705373
44       * <p>
45       * The method java.util.zip.ZipFile.open does not actually use the allocated
46       * memory of the heap, it maps the entire zip file into virtual memory
47       * outside the heap. If you run into this problem, you should try to reduce
48       * your heap size to about 600MB and try the restore again. This problem is
49       * allegedly fixed in java 6.
50       * 
51       * For now, using JarInputStream bypasses the memory mapped issues with
52       * JarFile, but may be worse performance.
53       * 
54       */
55  
56      public static void execute(File source, File destDir) throws IOException {
57          switch (readType) {
58          case JAR_FILE:
59              executeJarFile(source, destDir);
60              break;
61          case JAR_INPUT_STREAM:
62              executeJarInputStream(source, destDir);
63              break;
64          default:
65              throw new RuntimeException("Can't handle readtype: " + readType);
66          }
67  
68      }
69  
70      private static void executeJarFile(File source, File destDir) throws IOException {
71          destDir.mkdirs();
72          JarFile jf = new JarFile(source);
73          for (Enumeration<JarEntry> e = jf.entries(); e.hasMoreElements();) {
74              JarEntry je = e.nextElement();
75              System.out.println("Jar entry (a): " + je.getName() + " compressed: " + je.getCompressedSize() + " size: "
76                  + je.getSize() + " time: " + new Date(je.getTime()) + " comment: " + je.getComment());
77              java.io.File f = new java.io.File(destDir + java.io.File.separator + je.getName());
78  
79              if (je.isDirectory()) { // if its a directory, create it
80                  f.mkdir();
81                  continue;
82              }
83              try (java.io.InputStream is = new BufferedInputStream(jf.getInputStream(je))) {
84                  f.getParentFile().mkdirs();
85                  java.io.FileOutputStream fos = new java.io.FileOutputStream(f);
86                  byte[] buffer = new byte[102400];
87                  int bytesRead;
88                  while ((bytesRead = is.read(buffer, 0, 102400)) != -1) {
89                      // write contents of 'is' to 'fos'
90                      fos.write(buffer, 0, bytesRead);
91                  }
92                  fos.close();
93              }
94              f.setLastModified(je.getTime());
95          }
96      }
97  
98      private static void executeJarInputStream(File source, File destDir) throws IOException {
99          destDir.mkdirs();
100         FileInputStream fis = new FileInputStream(source);
101         BufferedInputStream bis = new BufferedInputStream(fis);
102         try (JarInputStream jis = new JarInputStream(bis)) {
103             JarEntry je = jis.getNextJarEntry();
104             while (je != null) {
105                 System.out.println("Jar entry (b): " + je.getName() + " compressed: " + je.getCompressedSize() + " size: "
106                     + je.getSize() + " time: " + new Date(je.getTime()) + " comment: " + je.getComment());
107                 java.io.File f = new java.io.File(destDir + java.io.File.separator + je.getName());
108                 if (je.isDirectory()) {
109                     f.mkdir();
110                 } else {
111                     f.getParentFile().mkdirs();
112                     try (java.io.FileOutputStream fos = new java.io.FileOutputStream(f)) {
113                         byte[] buffer = new byte[102400];
114                         int bytesRead;
115                         while ((bytesRead = jis.read(buffer, 0, 102400)) != -1) {
116                             // write contents of 'is' to 'fos'
117                             fos.write(buffer, 0, bytesRead);
118                         }
119                     }
120                     f.setLastModified(je.getTime());
121                 }
122                 je = jis.getNextJarEntry();
123             }
124         }
125     }
