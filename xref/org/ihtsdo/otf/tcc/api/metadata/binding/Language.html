70      public static ConceptVersionBI getConceptFromLang(

72              ViewCoordinate vc) throws IOException {
73          if (lang.toLowerCase().startsWith("en")) {
74              return EN.getStrict(vc);
75          }
76          if (lang.toLowerCase().startsWith("es")) {
77              return ES.getStrict(vc);
78          }
79          throw new UnsupportedOperationException("lang: " + lang);
