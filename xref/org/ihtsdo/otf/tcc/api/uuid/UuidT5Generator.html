38      public static UUID get(UUID namespace, String name) throws NoSuchAlgorithmException, UnsupportedEncodingException {
39          MessageDigest sha1Algorithm = MessageDigest.getInstance("SHA-1");
40  
41          // Generate the digest.
42          sha1Algorithm.reset();
43          if (namespace != null) {
44              sha1Algorithm.update(getRawBytes(namespace));
45          }
46          sha1Algorithm.update(name.getBytes(encoding));
47          byte[] sha1digest = sha1Algorithm.digest();
48  
49          sha1digest[6] &= 0x0f; /* clear version */
50          sha1digest[6] |= 0x50; /* set to version 5 */
51          sha1digest[8] &= 0x3f; /* clear variant */
52          sha1digest[8] |= 0x80; /* set to IETF variant */
53  
54          long msb = 0;
55          long lsb = 0;
56          for (int i = 0; i < 8; i++) {
57              msb = (msb << 8) | (sha1digest[i] & 0xff);
58          }
59          for (int i = 8; i < 16; i++) {
60              lsb = (lsb << 8) | (sha1digest[i] & 0xff);
61          }
62  
63          return new UUID(msb, lsb);
64      }
65  
66      public static UUID get(String name) throws NoSuchAlgorithmException, UnsupportedEncodingException {
67          return get(null, name);
68      }

77      /**
78       * This routine adapted from org.safehaus.uuid.UUID,
79       * which is licensed under Apache 2.
80       *
81       * @param uid
82       * @return
83       */
84      public static byte[] getRawBytes(UUID uid) {
85          String id = uid.toString();
86          byte[] rawBytes = new byte[16];
87  
88          for (int i = 0, j = 0; i < 36; ++j) {
89              // Need to bypass hyphens:
90              switch (i) {
91              case 8:
92              case 13:
93              case 18:
94              case 23:
95                  ++i;
96              }
97              char c = id.charAt(i);
98  
99              if (c >= '0' && c <= '9') {
100                 rawBytes[j] = (byte) ((c - '0') << 4);
101             } else if (c >= 'a' && c <= 'f') {
102                 rawBytes[j] = (byte) ((c - 'a' + 10) << 4);
103             }
104 
105             c = id.charAt(++i);
106 
107             if (c >= '0' && c <= '9') {
108                 rawBytes[j] |= (byte) (c - '0');
109             } else if (c >= 'a' && c <= 'f') {
110                 rawBytes[j] |= (byte) (c - 'a' + 10);
111             }
112             ++i;
113         }
114         return rawBytes;
115     }

118     /**
119      * Generates a uuid from the given <code>byteArray</code>.
120      *
121      * @param byteArray the bytes to use for generating the uuid
122      * @return the generated uuid
123      */
124     public static UUID getUuidFromRawBytes(byte[] byteArray) {
125         if (byteArray.length != 16) {
126             throw new NumberFormatException("UUID must be 16 bytes");
127         }
128 
129         ByteBuffer raw = ByteBuffer.wrap(byteArray);
130 
131         return new UUID(raw.getLong(raw.position()), raw.getLong(raw.position() + 8));
132     }
