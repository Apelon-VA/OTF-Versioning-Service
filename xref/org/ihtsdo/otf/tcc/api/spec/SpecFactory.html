16      public static ConceptSpec get(ConceptChronicleBI concept, ViewCoordinate vc) throws IOException {
17          ConceptVersionBI cv = Ts.get().getConceptVersion(vc, concept.getNid());
18          try {
19              return new ConceptSpec(cv.getDescriptionsActive().iterator().next().getText(),
20                      concept.getPrimordialUuid());
21          } catch (ContradictionException ex) {
22              for (DescriptionChronicleBI desc : concept.getDescriptions()) {
23                  for (DescriptionVersionBI dv : desc.getVersions(vc)) {
24                      return new ConceptSpec(dv.getText(),
25                              concept.getPrimordialUuid());
26                  }
27              }
28              throw new IOException("No current description for: " + cv);
29          }
30      }

33          try {

35                      concept.getPrimordialUuid());

43      public static DescriptionSpec get(DescriptionVersionBI desc, ViewCoordinate vc) throws IOException {
44          if (desc != null && desc.getUUIDs() != null) {
45              DescriptionSpec ds = new DescriptionSpec(desc.getUUIDs().toArray(new UUID[]{}),
46                      get(Ts.get().getConcept(desc.getConceptNid()), vc),
47                      get(Ts.get().getConcept(desc.getTypeNid()), vc),
48                      desc.getText());
49              ds.setLangText(desc.getText());
