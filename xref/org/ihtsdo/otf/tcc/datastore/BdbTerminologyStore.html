80      public BdbTerminologyStore() {
81          if (databaseSetup.compareAndSet(false, true)) {
82              bdbLocation = System.getProperty(BDB_LOCATION_PROPERTY);
83              if (bdbLocation == null) {
84                  bdbLocation = "berkeley-db";
85                  LOG.info(BDB_LOCATION_PROPERTY
86                          + " not set. Using default location of: " + DEFAULT_BDB_LOCATION);
87              } else {
88                  LOG.log(Level.INFO, BDB_LOCATION_PROPERTY
89                          + " set. Starting from location: {0}", bdbLocation);
90              }
91              Bdb.setup(bdbLocation, this);
92              LOG.info("Database setup complete");
93              setupComplete.countDown();
94          } else {
95              LOG.info("Database setup already initialized");
96          }
97          try {
98              setupComplete.await();
99          } catch (InterruptedException ex) {
100             Logger.getLogger(BdbTerminologyStore.class.getName()).log(Level.SEVERE, null, ex);
101         }
102     }

271         try {
272             LinkedBlockingQueue<ConceptConverter> converters = new LinkedBlockingQueue<>();
273             int runtimeConverterSize =

277             AtomicInteger conceptsProcessed = new AtomicInteger();

281                 converters.add(new ConceptConverter(converters, conceptsProcessed, indexedAnnotationConcepts));

390     @Override
391     public boolean satisfiesDependencies(Collection<DbDependency> dependencies) {
392         if (dependencies != null) {
393             try {
394                 for (DbDependency d : dependencies) {
395                     String value = P.s.getProperty(d.getKey());
396 
397                     if (d.satisfactoryValue(value) == false) {
398                         return false;
399                     }
400                 }
401             } catch (Throwable e) {
402                 AceLog.getAppLog().alertAndLogException(e);
403 
404                 return false;
405             }
406         }
407 
408         return true;
409     }

555             CsProperty.LAST_CHANGE_SET_READ};

853     private static class ConceptConverter implements Runnable {
854 
855         TtkConceptChronicle eConcept = null;
856         Throwable exception = null;
857         ConceptChronicle newConcept = null;
858         AtomicInteger conceptsProcessed;
859         LinkedBlockingQueue<ConceptConverter> converters;
860         ConcurrentSkipListSet<ConceptChronicleBI> indexedAnnotationConcepts;
861         NidCNidMapBdb nidCnidMap;
862 
863         public ConceptConverter(LinkedBlockingQueue<ConceptConverter> converters, AtomicInteger conceptsRead,
864                 ConcurrentSkipListSet<ConceptChronicleBI> indexedAnnotationConcepts) {
865             this.converters = converters;
866             this.conceptsProcessed = conceptsRead;
867             this.indexedAnnotationConcepts = indexedAnnotationConcepts;
868         }
869 
870         @Override
871         public void run() {
872             if (nidCnidMap == null) {
873                 nidCnidMap = Bdb.getNidCNidMap();
874             }
875 
876             try {
877                 newConcept = ConceptChronicle.get(eConcept, indexedAnnotationConcepts);
878 
879                 if (newConcept != null) {
880                     assert newConcept.readyToWrite();
881                     Bdb.getConceptDb().writeConcept(newConcept);
882 
883                     Collection<Integer> nids = newConcept.getAllNids();
884 
885                     assert nidCnidMap.getCNid(newConcept.getNid()) == newConcept.getNid();
886 
887                     for (int nid : nids) {
888                         assert nidCnidMap.getCNid(nid) == newConcept.getNid();
889                     }
890                 }
891 
892                 conceptsProcessed.incrementAndGet();
893             } catch (Throwable e) {
894                 exception = e;
895             }
896 
897             converters.add(this);
898         }
