76      public PrimitiveIntObjectMap(final int tableSize,
77                              final int indexIntervals) {
78          // determine number of shifts for intervals
79          int i = 1;
80          int size = 2;
81          while ( size < indexIntervals ) {
82              size <<= 1;
83              ++i;
84          }
85          this.indexIntervals = size;
86          this.intervalShifts = i;
87  
88          // determine number of shifts for tableSize
89          i = 1;
90          size = 2;
91          while ( size < tableSize ) {
92              size <<= 1;
93              ++i;
94          }
95          this.tableSize = size;
96          this.shifts = i;
97          this.doubleShifts = this.shifts << 1;
98  
99          // determine mid point of an interval
100         this.midIntervalPoint = ((this.tableSize << this.shifts) << this.intervalShifts) >> 1;
101 
102         this.lastPageId = 0;
103 
104         init();
105     }

162     public Object put(final int key,
163     		Object value) {
164         if ( key < 0 ) {
165             throw new IllegalArgumentException( "-ve keys not supported: " + key );
166         }
167 
168         // NULL is a placeholder to show the key exists
169         // but contains a null value
170         if ( value == null ) {
171             value = PrimitiveIntObjectMap.NULL;
172         }
173 
174         final Page page = findPage( key );
175 
176         final Object oldValue = page.put( key,value );
177 
178         if ( oldValue == null ) {
179             this.totalSize++;
180         }
181 
182         return oldValue;
183     }

185     public Object remove(final int key) {
186         if ( key > this.maxKey || key < 0 ) {
187             return null;
188         }
189 
190         final Page page = findPage( key );
191 
192         final Object oldValue = page.put( key,null );
193 
194         if ( this.lastPageId != 0 && this.lastPage.isEmpty() ) {
195             shrinkPages( this.lastPageId );
196         }
197 
198         if ( oldValue != null ) {
199             this.totalSize--;
200         }
201 
202         return oldValue;
203     }

205     public Object get(final int key) {
206         if ( key > this.maxKey || key < 0 ) {
207             return null;
208         }
209 
210         Object value = findPage( key ).get( key );
211 
212         // NULL means the key exists, so return a real null
213         if ( value == PrimitiveIntObjectMap.NULL ) {
214             value = null;
215         }
216         return value;

219     /**
220      * gets the next populated key, after the given key position.
221      * @param key
222      * @return
223      */
224     public int getNext(int key) {
225         final int currentPageId = key >> this.doubleShifts;
226         final int nextPageId = (key+1) >> this.doubleShifts;
227 
228         if ( currentPageId != nextPageId ) {
229             Page page = findPage( key + 1);
230             while ( page.isEmpty() ) {
231                 page = page.getNextSibling();
232             }
233             key = this.doubleShifts << page.getPageId();
234         } else {
235             key += 1;
236         }
237 
238         while ( !containsKey( key ) && key <= this.maxKey ) {
239             key++;
240         }
241 
242         if ( key > this.maxKey ) {
243             key -= 1;
244         }
245 
246         return key;
247     }

254     public Collection values() {
255         final CompositeCollection collection = new CompositeCollection();
256         Page page = this.firstPage;
257 
258         while ( page != null && page.getPageId() <= this.lastPageId ) {
259             collection.addComposited( Arrays.asList( page.getValues() ) );
260             page = page.getNextSibling();
261         }

273     /**
274      * Expand index to accomodate given pageId Create empty TopNodes
275      */
276     public Page expandPages(final int toPageId) {
277         for ( int x = this.lastPageId; x < toPageId; x++ ) {
278             this.lastPage = new Page( this.lastPage,
279                                       ++this.lastPageId,
280                                       this.tableSize );
281             // index interval, so expand index
282             if ( this.lastPage.getPageId() % this.indexIntervals == 0 ) {
283                 final int newSize = this.pageIndex.length + 1;
284                 resizeIndex( newSize );
285                 this.pageIndex[newSize - 1] = this.lastPage;
286             }
287         }
288         this.maxKey = (this.lastPageId + 1 << this.doubleShifts) - 1;
289         return this.lastPage;
290     }

292     /**
293      * Shrink index to accomodate given pageId
294      */
295     public void shrinkPages(final int toPageId) {
296         for ( int x = this.lastPageId; x >= toPageId; x-- ) {
297             // last page is on index so shrink index
298             if ( (this.lastPageId) % this.indexIntervals == 0 && this.lastPageId != 0 ) {
299                 resizeIndex( this.pageIndex.length - 1 );
300             }
301 
302             final Page page = this.lastPage.getPreviousSibling();
303             page.setNextSibling( null );
304             this.lastPage.clear();
305             this.lastPage = page;
306             this.lastPageId = page.getPageId();
307 
308         }
309     }

311     public void resizeIndex(final int newSize) {
312         final Page[] newIndex = new Page[newSize];
313         System.arraycopy( this.pageIndex,
314                           0,
315                           newIndex,
316                           0,
317                           (newSize > this.pageIndex.length) ? this.pageIndex.length : newSize );
318         this.pageIndex = newIndex;
319     }

321     private Page findPage(final int key) {
322         // determine Page
323         final int pageId = key >> this.doubleShifts;
324         Page page;
325 
326         // if pageId is lastNodeId use lastNode reference
327         if ( pageId == this.lastPageId ) {
328             page = this.lastPage;
329         }
330         // if pageId is zero use first page reference
331         else if ( pageId == 0 ) {
332             page = this.firstPage;
333         }
334         // if pageId is greater than lastTopNodeId need to expand
335         else if ( pageId > this.lastPageId ) {
336             page = expandPages( pageId );
337         } else {
338             // determine offset
339             final int offset = pageId >> this.intervalShifts;
340             // are we before or after the halfway point of an index interval
341             if ( (offset != (this.pageIndex.length - 1)) && ((key - (offset << this.intervalShifts << this.doubleShifts)) > this.midIntervalPoint) ) {
342                 // after so go to next node index and go backwards
343                 page = this.pageIndex[offset + 1];
344                 while ( page.getPageId() != pageId ) {
345                     page = page.getPreviousSibling();
346                 }
347             } else {
348                 // before so go to node index and go forwards
349                 page = this.pageIndex[offset];
350                 while ( page.getPageId() != pageId ) {
351                     page = page.getNextSibling();
352                 }
353             }
354         }
355 
356         return page;
357     }

446         public Object get(int key) {
447             if ( this.tables == null ) {
448                 return null;
449             }
450             // normalise key
451             key -= this.pageSize * this.pageId;
452 
453             // determine table
454             final int table = key >> this.shifts;
455 
456             // determine offset
457             final int offset = table << this.shifts;
458 
459             // tables[table][slot]
460             return this.tables[table][key - offset];
461         }

463         public Object put(int key, final Object newValue) {
464             if ( this.tables == null ) {
465                 // initiate tree;
466                 this.tables = new Object[this.tableSize][this.tableSize];
467             }
468 
469             // normalise key
470             key -= this.pageSize * this.pageId;
471 
472             // determine table
473             final int table = key >> this.shifts;
474 
475             // determine offset
476             final int offset = table << this.shifts;
477 
478             // determine slot
479             final int slot = key - offset;
480 
481             // get old value
482             final Object oldValue = this.tables[table][slot];
483             this.tables[table][slot] = newValue;
484 
485             // update number of empty cells for TopNode
486             if ( oldValue == null && newValue != null ) {
487                 this.filledSlots++;
488             } else if ( oldValue != null && newValue == null ) {
489                 this.filledSlots--;
490             }
491 
492             // if this page contains no values then null the array
493             // to allow it to be garbage collected
494             if ( this.filledSlots == 0 ) {
495                 this.tables = null;
496             }
497 
498             return oldValue;
499         }

505         Object[] getValues() {
506             final Object[] values = new Object[this.filledSlots];
507             if ( values.length == 0 ) {
508                 return values;
509             }
510             int x = 0;
511             Object value;
512             for ( int i = 0; i < this.tableSize; i++ ) {
513                 for ( int j = 0; j < this.tableSize; j++ ) {
514                     value = this.tables[i][j];
515                     if ( value != null ) {
516                         // swap NULL out placeholder
517                         // Also filter out InitialFact
518                         if ( value == PrimitiveIntObjectMap.NULL ) {
519                             value = null;
520                         }
521                         values[x] = value;
522                         x++;
523                     }
524                 }
525             }
526             return values;
527         }
