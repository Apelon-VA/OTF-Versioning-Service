117     public static void setCacheSize(String cacheSize) {
118         long size;
119         if (cacheSize.toLowerCase().endsWith("m")) {
120             cacheSize = cacheSize.substring(0, cacheSize.length() - 1);
121             size = Integer.parseInt(cacheSize);
122             size = size * 1000;
123         } else if (cacheSize.toLowerCase().endsWith("g")) {
124             cacheSize = cacheSize.substring(0, cacheSize.length() - 1);
125             size = Integer.parseInt(cacheSize);
126             size = size * 1000000;
127         } else {
128             size = Integer.parseInt(cacheSize);
129         }
130         EnvironmentMutableConfig mutableConfig = Bdb.mutable.bdbEnv.getMutableConfig();
131         mutableConfig.setCacheSize(size);
132         Bdb.mutable.bdbEnv.setMutableConfig(mutableConfig);
133         Bdb.readOnly.bdbEnv.setMutableConfig(mutableConfig);
134     }

517     public static void setup(String dbRoot, boolean staticPublish) {
518 
519         System.out.println("setup dbRoot: " + dbRoot);
520         stampCache = new ConcurrentHashMap<>();
521         try {
522             closed = false;
523             syncService = Executors.newFixedThreadPool(1,
524                     new NamedThreadFactory(dbdThreadGroup, "Sync service"));
525 
526             BdbCommitManager.reset();
527             NidDataFromBdb.resetExecutorPool();
528             BdbPathManager.reset();
529 
530             for (@SuppressWarnings("unused") OFFSETS o : OFFSETS.values()) {
531                 // ensure all OFFSETS are initialized prior to multi-threading. 
532             }
533             bdbDirectory = new File(dbRoot);
534             System.out.println("absolute dbRoot: " + bdbDirectory.getAbsolutePath());
535             viewCoordinateMapFile = new File(bdbDirectory, "viewCoordinates.oos");
536             bdbDirectory.mkdirs();
537             LuceneManager.setLuceneRootDir(bdbDirectory);
538 
539             mutable = new Bdb(false, new File(bdbDirectory, "mutable"));
540             File readOnlyDir = new File(bdbDirectory, "read-only");
541             boolean readOnlyExists = readOnlyDir.exists();
542             readOnly = new Bdb(readOnlyExists, readOnlyDir);
543             if (Looker.lookup(TtkEnvironment.class).useFxWorkers()) {
544 
545                 Platform.runLater(new Runnable() {
546                     @Override
547                     public void run() {
548                         try {
549 
550                             startupWithFx();
551 
552                         } catch (InterruptedException | ExecutionException ex) {
553                             Logger.getLogger(Bdb.class.getName()).log(Level.SEVERE, null, ex);
554                         }
555                     }
556                 });
557             } else {
558                 startupNoFx();
559             }
560 
561 
562         } catch (IOException | DatabaseException | IllegalArgumentException e) {
563             throw new RuntimeException(e);
564         } catch (InterruptedException ex) {
565             Logger.getLogger(Bdb.class.getName()).log(Level.SEVERE, null, ex);
566         } catch (ExecutionException ex) {
567             Logger.getLogger(Bdb.class.getName()).log(Level.SEVERE, null, ex);
568         }
569     }

571     private static void finishSetup() {
572         setupLatch.countDown();
573         if (setupLatch.getCount() == 0) {
574             try {
575                 BdbTerminologyStore ts = new BdbTerminologyStore();
576                 if (P.s == null) {
577                     Ts.set(ts);
578                     FxTs.set(ts);
579                     P.s = ts;
580                 }
581                 Looker.add(ts, UUID.randomUUID(), "Embedded BdbTerminologyStore");
582                 Looker.add(new TtkConceptChronicleConverter(), 
583                         UUID.randomUUID(), "TtkConceptChronicle to ConceptChronicleDTO converter");
584                 ConceptChronicle.reset();
585 
586                 ReferenceConcepts.reset();
587                 String versionString = getProperty(G_VERSION);
588                 if (versionString != null) {
589                     gVersion.set(Long.parseLong(versionString));
590                 }
591                 if (viewCoordinateMapFile.exists()) {
592                     FileInputStream fis = new FileInputStream(viewCoordinateMapFile);
593                     try (ObjectInputStream ois = new ObjectInputStream(fis)) {
594                         try {
595                             viewCoordinates = (ConcurrentHashMap<UUID, ViewCoordinate>) ois.readObject();
596                         } catch (IOException | ClassNotFoundException ex) {
597                             Logger.getLogger(Bdb.class.getName()).log(Level.SEVERE, null, ex);
598                         }
599                     }
600                 }
601                 pathManager = BdbPathManager.get();
602                 Looker.lookup(TermstoreLatch.class).openLatch();
603 
604 
605 
606                 AceLog.getAppLog().info("mutable maxMem: "
607                         + Bdb.mutable.bdbEnv.getConfig().getConfigParam("je.maxMemory"));
608                 AceLog.getAppLog().info("mutable shared cache: "
609                         + Bdb.mutable.bdbEnv.getConfig().getSharedCache());
610                 AceLog.getAppLog().info("readOnly maxMem: "
611                         + Bdb.readOnly.bdbEnv.getConfig().getConfigParam("je.maxMemory"));
612                 AceLog.getAppLog().info("readOnly shared cache: "
613                         + Bdb.readOnly.bdbEnv.getConfig().getSharedCache());
614 
615             } catch (IOException ex) {
616                 Logger.getLogger(Bdb.class.getName()).log(Level.SEVERE, null, ex);
617             }
618         }
619     }
