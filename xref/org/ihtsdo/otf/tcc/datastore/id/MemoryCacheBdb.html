43  /**
44   * Caches a variety of relationships of a component to other components as
45   * defined by nids in arrays kept in memory for performance.
46   * <h2>Implementation Details</h2> The
47   * <code>nid</code> is the
48   * <code>(nid - Integer.MIN_VALUE)</code> index into an
49   * <code>int[][]</code> which stores sequentially increasing arrays of cNid.
50   * <br> <br>This array approach is taken because Java stores multidimensional
51   * arrays as arrays of arrays, rather than a contiguous block of arrays. Each
52   * array has an overhead of 96 bits (above and beyond its data), which doubles
53   * the memory size, and also increases the burden on the garbage collector.
54   *
55   *
56   * @author kec
57   *
58   */
59  public class MemoryCacheBdb extends ComponentBdb {

73      public void addNidPairForRefex(int nid, NidPairForRefex pair) throws IOException {
74          int mapIndex = (nid - Integer.MIN_VALUE) / NID_CNID_MAP_SIZE;
75          int nidIndexInMap = ((nid - Integer.MIN_VALUE) % NID_CNID_MAP_SIZE);
76  
77          assert (mapIndex >= 0) && (nidIndexInMap >= 0) :
78                  "mapIndex: " + mapIndex + " indexInMap: " + nidIndexInMap + " nid: " + nid;
79  
80          if (mapIndex >= nidCNidMaps.get().length) {
81              ensureCapacity(nid);
82          }
83  
84          IndexCacheRecord record = new IndexCacheRecord(indexCacheRecords.get()[mapIndex][nidIndexInMap]);
85  
86          if (!record.isRefexMemberAlreadyThere(pair.getMemberNid())) {
87              locks.lock(nid);
88              try {
89                  record = new IndexCacheRecord(indexCacheRecords.get()[mapIndex][nidIndexInMap]);
90                  record.addNidPairForRefex(pair.getRefexNid(), pair.getMemberNid());
91                  indexCacheRecords.get()[mapIndex][nidIndexInMap] = record.getData();
92                  mapChanged[mapIndex] = true;
93              } finally {
94                  locks.unlock(nid);
95              }
96          }
97      }

138     private void ensureCapacity(int nextId) throws IOException {
139         long next = nextId;
140         long numIds = (next - Integer.MIN_VALUE);
141         long maps = numIds / NID_CNID_MAP_SIZE;
142         int nidCidMapCount = (int) maps + 1;
143 
144         rwl.readLock().lock();
145 
146         try {
147             if (nidCidMapCount > nidCNidMaps.get().length) {
148                 rwl.readLock().unlock();
149                 rwl.writeLock().lock();
150 
151                 if (nidCidMapCount > nidCNidMaps.get().length) {
152                     try {
153                         expandCapacity(nidCidMapCount);
154                     } finally {
155                         rwl.readLock().lock();
156                         rwl.writeLock().unlock();
157                     }
158                 } else {
159                     rwl.readLock().lock();
160                     rwl.writeLock().unlock();
161                 }
162             }
163         } finally {
164             rwl.readLock().unlock();
165         }
166     }

168     private void expandCapacity(int nidCidMapCount) throws IOException {
169         int oldCount = nidCNidMaps.get().length;
170         int[][] newNidCidMaps = new int[nidCidMapCount][];
171         int[][][] newIndexCacheRecords = new int[nidCidMapCount][][];
172         boolean[] newMapChanged = new boolean[nidCidMapCount];
173 
174         for (int i = 0; i < oldCount; i++) {
175             newIndexCacheRecords[i] = indexCacheRecords.get()[i];
176             newNidCidMaps[i] = nidCNidMaps.get()[i];
177             newMapChanged[i] = mapChanged[i];
178         }
179 
180         for (int i = oldCount; i < nidCidMapCount; i++) {
181             newIndexCacheRecords[i] = new int[NID_CNID_MAP_SIZE][];
182             newNidCidMaps[i] = new int[NID_CNID_MAP_SIZE];
183             newMapChanged[i] = true;
184             Arrays.fill(newNidCidMaps[i], Integer.MAX_VALUE);
185         }
186 
187         indexCacheRecords.set(newIndexCacheRecords);
188         nidCNidMaps.set(newNidCidMaps);
189         mapChanged = newMapChanged;
190     }

192     public void forgetNidPairForRefex(int nid, NidPairForRefex pair) {
193         int mapIndex = (nid - Integer.MIN_VALUE) / NID_CNID_MAP_SIZE;
194         int nidIndexInMap = ((nid - Integer.MIN_VALUE) % NID_CNID_MAP_SIZE);
195 
196         assert (mapIndex >= 0) && (nidIndexInMap >= 0) :
197                 "mapIndex: " + mapIndex + " indexInMap: " + nidIndexInMap + " nid: " + nid;
198 
199         if (mapIndex >= nidCNidMaps.get().length) {
200             throw new NoSuchElementException("nid: " + nid);
201         }
202 
203         locks.lock(nid);
204 
205         try {
206             IndexCacheRecord record = new IndexCacheRecord(indexCacheRecords.get()[mapIndex][nidIndexInMap]);
207 
208             record.forgetNidPairForRefex(pair.getRefexNid(), pair.getMemberNid());
209             indexCacheRecords.get()[mapIndex][nidIndexInMap] = record.getData();
210             mapChanged[mapIndex] = true;
211         } finally {
212             locks.unlock(nid);
213         }

217     protected void init() throws IOException {
218         preloadBoth();
219 
220         int maxId = Bdb.getUuidsToNidMap().getCurrentMaxNid();
221 
222         readOnlyRecords = (int) readOnly.count();
223 
224         int mutableRecords = (int) mutable.count();
225 
226         AceLog.getAppLog().info("NidCidMap readOnlyRecords: " + readOnlyRecords);
227         AceLog.getAppLog().info("NidCidMap mutableRecords: " + mutableRecords);
228 
229         int nidCidMapCount = ((maxId - Integer.MIN_VALUE) / NID_CNID_MAP_SIZE) + 1;
230 
231         nidCNidMaps = new AtomicReference<>(new int[nidCidMapCount][]);
232         indexCacheRecords = new AtomicReference<>(new int[nidCidMapCount][][]);
233         mapChanged = new boolean[nidCidMapCount];
234         Arrays.fill(mapChanged, false);
235 
236         for (int index = 0; index < nidCidMapCount; index++) {
237             nidCNidMaps.get()[index] = new int[NID_CNID_MAP_SIZE];
238             indexCacheRecords.get()[index] = new int[NID_CNID_MAP_SIZE][];
239             Arrays.fill(nidCNidMaps.get()[index], Integer.MAX_VALUE);
240         }
241 
242         readMaps(readOnly);
243         closeReadOnly();
244         readMaps(mutable);
245 
246         if (AceLog.getAppLog().isLoggable(Level.FINE)) {
247             printKeys("Read only keys: ", readOnly);
248             printKeys("Mutable keys: ", mutable);
249         }
250 
251         rwl = new ReentrantReadWriteLock();
252     }

254     private void printKeys(String prefix, Database db) {
255         int size = (int) db.count();
256         OpenIntIntHashMap nidMap = new OpenIntIntHashMap(size + 2);
257         CursorConfig cursorConfig = new CursorConfig();
258 
259         cursorConfig.setReadUncommitted(true);
260 
261         try (Cursor cursor = db.openCursor(null, cursorConfig)) {
262             DatabaseEntry foundKey = new DatabaseEntry();
263             DatabaseEntry foundData = new DatabaseEntry();
264 
265             foundData.setPartial(true);
266             foundData.setPartial(0, 0, true);
267 
268             int max = Integer.MIN_VALUE;
269 
270             while (cursor.getNext(foundKey, foundData, LockMode.READ_UNCOMMITTED) == OperationStatus.SUCCESS) {
271                 int cNid = IntegerBinding.entryToInt(foundKey);
272 
273                 nidMap.put(cNid, cNid);
274                 max = Math.max(max, cNid);
275             }
276 
277             cursor.close();
278             AceLog.getAppLog().fine(prefix + nidMap.keys().toList().toString());
279         }
280     }

282     private void readMaps(Database db) {
283         CursorConfig cursorConfig = new CursorConfig();
284 
285         cursorConfig.setReadUncommitted(true);
286 
287         try (Cursor cursor = db.openCursor(null, cursorConfig)) {
288             DatabaseEntry foundKey = new DatabaseEntry();
289             DatabaseEntry foundData = new DatabaseEntry();
290 
291             while (cursor.getNext(foundKey, foundData, LockMode.READ_UNCOMMITTED) == OperationStatus.SUCCESS) {
292                 int index = IntegerBinding.entryToInt(foundKey);
293                 TupleInput ti = new TupleInput(foundData.getData());
294                 int j = 0;
295 
296                 while (ti.available() > 0) {
297                     nidCNidMaps.get()[index][j] = ti.readInt();
298 
299                     int length = ti.readSortedPackedInt();
300 
301                     if (length != 0) {
302                         int[] cacheData = new int[length];
303 
304                         for (int i = 0; i < length; i++) {
305                             cacheData[i] = ti.readInt();
306                         }
307 
308                         indexCacheRecords.get()[index][j] = cacheData;
309                     }
310 
311                     j++;
312                 }
313             }
314         }
315     }

317     public void resetCNidForNid(int cNid, int nid) throws IOException {
318         assert cNid != Integer.MAX_VALUE;
319 
320         int mapIndex = (nid - Integer.MIN_VALUE) / NID_CNID_MAP_SIZE;
321 
322         assert mapIndex >= 0 : "cNid: " + cNid + " nid: " + nid + " mapIndex: " + mapIndex;
323 
324         int cNidIndexInMap = (nid - Integer.MIN_VALUE) % NID_CNID_MAP_SIZE;
325 
326         assert cNidIndexInMap < NID_CNID_MAP_SIZE :
327                 "cNid: " + cNid + " nid: " + nid + " mapIndex: " + mapIndex + " cNidIndexInMap: " + cNidIndexInMap;
328         ensureCapacity(nid);
329 
330         if ((nidCNidMaps.get() != null) && (nidCNidMaps.get()[mapIndex] != null)) {
331             if (nidCNidMaps.get()[mapIndex][cNidIndexInMap] != cNid) {
332                 if (cNid < 0) {
333                     nidCNidMaps.get()[mapIndex][cNidIndexInMap] = cNid * -1;
334                 } else {
335                     nidCNidMaps.get()[mapIndex][cNidIndexInMap] = cNid;
336                 }
337                 mapChanged[mapIndex] = true;
338             }
339         } else {
340             if (nidCNidMaps.get() == null) {
341                 throw new IOException("Null nidCidMap: ");
342             }
343 
344             throw new IOException("nidCidMap[" + mapIndex + "] " + "is null. cNid: " + cNid + " nid: " + nid);
345         }
346     }

354     public void updateOutgoingRelationshipData(ConceptChronicle concept) throws IOException {
355         int cNid = concept.getNid();
356         int mapIndex = (cNid - Integer.MIN_VALUE) / NID_CNID_MAP_SIZE;
357         int nidIndexInMap = ((cNid - Integer.MIN_VALUE) % NID_CNID_MAP_SIZE);
358 
359         assert (mapIndex >= 0) && (nidIndexInMap >= 0) :
360                 "mapIndex: " + mapIndex + " indexInMap: " + nidIndexInMap + " cNid: " + cNid;
361 
362         if (mapIndex >= nidCNidMaps.get().length) {
363             ensureCapacity(cNid);
364         }
365 
366         locks.lock(concept.getNid());
367 
368         try {
369             IndexCacheRecord record =
370                     new IndexCacheRecord(indexCacheRecords.get()[mapIndex][nidIndexInMap]);
371             int[] indexCacheRecordRelationshipArray = RelationshipIndexRecordFactory.make(concept);
372 
373             record.updateData(indexCacheRecordRelationshipArray, record.getDestinationOriginNids(),
374                     record.getRefexIndexArray());
375             indexCacheRecords.get()[mapIndex][nidIndexInMap] = record.getData();
376             mapChanged[mapIndex] = true;
377         } finally {
378             locks.unlock(concept.getNid());
379         }

382     private void writeChangedMaps() throws IOException {
383         rwl.writeLock().lock();
384 
385         try {
386             DatabaseEntry keyEntry = new DatabaseEntry();
387             int mapCount = nidCNidMaps.get().length;
388 
389             found:
390             for (int key = 0; key < mapCount; key++) {
391                 if (mapChanged[key]) {
392                     IntegerBinding.intToEntry(key, keyEntry);
393 
394                     // NID_CNID_MAP_SIZE * 4 * 2 = 4 bytes per integer, 2 integers per nid
395                     TupleOutput output = new TupleOutput(new byte[NID_CNID_MAP_SIZE * 4 * 2]);
396 
397                     for (int i = 0; i < NID_CNID_MAP_SIZE; i++) {
398                         output.writeInt(nidCNidMaps.get()[key][i]);
399 
400                         int[] cacheRecord = indexCacheRecords.get()[key][i];
401 
402                         if (cacheRecord == null) {
403                             output.writeSortedPackedInt(0);
404                         } else {
405                             output.writeSortedPackedInt(cacheRecord.length);
406 
407                             for (int j = 0; j < cacheRecord.length; j++) {
408                                 output.writeInt(cacheRecord[j]);
409                             }
410                         }
411                     }
412 
413                     DatabaseEntry valueEntry = new DatabaseEntry(output.toByteArray());
414                     OperationStatus status = mutable.put(null, keyEntry, valueEntry);
415 
416                     if (status != OperationStatus.SUCCESS) {
417                         throw new IOException("Unsuccessful operation: " + status);
418                     }
419 
420                     mapChanged[key] = false;
421                 }
422             }
423         } finally {
424             rwl.writeLock().unlock();
425         }
426     }

550      * Retrieves the components nids enclosed by the input conceptNids.
551      *
552      * @param conceptNativeIds the <code>NativeIdSetBI</code> for which the
553      * components nids will be retrieved
554      */
555     public NativeIdSetBI getComponentNidsForConceptNids(NativeIdSetBI conceptNids) throws IOException {
556         NativeIdSetBI componentNids = new ConcurrentBitSet();
557         int maxNid = Bdb.getUuidsToNidMap().getCurrentMaxNid();
558         for (int i = Integer.MIN_VALUE; i <= maxNid; i++) {
559             int mapIndex = (i - Integer.MIN_VALUE) / NID_CNID_MAP_SIZE;
560             int cNidIndexInMap = ((i - Integer.MIN_VALUE) % NID_CNID_MAP_SIZE);
561             int nid = nidCNidMaps.get()[mapIndex][cNidIndexInMap];
562             if (nid > 0) {
563                 nid = nid * -1;
564             }
565             if (conceptNids.contains(nid)) {
566                 componentNids.add(i);
567             }
568         }
569         return componentNids;

573     public boolean hasConcept(int cNid) {
574         assert cNid > Integer.MIN_VALUE : "Invalid cNid == Integer.MIN_VALUE: " + cNid;
575         assert cNid <= Bdb.getUuidsToNidMap().getCurrentMaxNid() :
576                 "Invalid cNid: " + cNid + " currentMax: " + Bdb.getUuidsToNidMap().getCurrentMaxNid();
577 
578         int mapIndex = (cNid - Integer.MIN_VALUE) / NID_CNID_MAP_SIZE;
579         int cNidIndexInMap = ((cNid - Integer.MIN_VALUE) % NID_CNID_MAP_SIZE);
580 
581         if ((mapIndex < 0) || (mapIndex >= nidCNidMaps.get().length)) {
582             return false;
583         }
584 
585         if ((cNidIndexInMap < 0) || (cNidIndexInMap >= NID_CNID_MAP_SIZE)) {
586             return false;
587         }
588 
589         int cNid2 = nidCNidMaps.get()[mapIndex][cNidIndexInMap];
590 
591         if (cNid2 == cNid) {
592             return true;
593         }
594         if (cNid2 == cNid * -1) {
595             return true;
596         }
597 
598         return false;
599     }
