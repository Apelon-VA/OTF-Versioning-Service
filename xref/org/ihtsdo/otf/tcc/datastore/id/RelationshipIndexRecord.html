33  /**
34   * Origin relationship data stores the following data for each entry:
35   * <br>
36   * <br><code>[record length]</code>
37   * <br><code>[typeNid]</code>
38   * <br><code>[destinationNid]</code>
39   * <br><code>[inferredBit + groupGtZeroBit + STAMP]</code>
40   * <br><code>[groupId]</code> Only if indicated by groupGtZeroBit
41   * <br><code>[inferredBit + groupGtZeroBit + STAMP]</code> addition record only if > 1 STAMP
42   * <br><code>[groupId]</code> Only if indicated by groupGtZeroBit
43   * <br><code>[inferredBit + groupGtZeroBit + STAMP]</code> addition record only if > 2 STAMP
44   * <br><code>[groupId]</code> Only if indicated by groupGtZeroBit
45   * <br>...
46   * <br> Note this record has no member id. If inferred and stated relationships have the
47   * same type, the STAMPs and groups are merged into a single record. This combination of stated and inferred
48   * saves 96 bits/inferred relationship, and makes search more efficient, since there is less data to
49   * search over. In addition, the compression of STAMP, Group, and Inferred values for relationships in group 0
50   * saves an additional 64 bits, for ~196 bit savings per relationship. Further savings are also realized by
51   * not requiring each revision to represent the type and refinability, and eliminating an object pointer
52   * for each revision (96 bits + 64 bits per revision).
53   * @author kec
54   */
55  public class RelationshipIndexRecord implements Iterable<RelationshipIndexRecord> {
56     private static final int   DESTINATION_NID_OFFSET          = 2;
57     protected static final int GROUP_BITMASK                   = (1 << 30);
58     protected static final int INFERRED_BITMASK                = (1 << 31);
59     private static final int   RECORD_LENGTH_OFFSET            = 0;
60     private static final int   STAMP_INFERRED_HAS_GROUP_OFFSET = 3;
61     private static final int   TYPE_NID_OFFSET                 = 1;
62     private static int         inferredNid                     = Integer.MAX_VALUE;
63     private static int         statedNid                       = Integer.MAX_VALUE;

81     /**
82      * Allows iteration over the relationship records without allocating an object for each
83      * record, it just reuses this object, but updates the offset to point to the next record.
84      * @return this <code>RelationshipIndexRecord</code> with the offset of the next record,
85      * or null if the end of the relationship data is
86      */
87     public RelationshipIndexRecord advance() {
88        if (hasNext()) {
89           offset = offset + data[offset + RECORD_LENGTH_OFFSET];
90  
91           return this;
92        }
93  
94        throw new NoSuchElementException();
95     }

112    public List<RelationshipIndexVersion> getVersions() throws IOException {
113       if (inferredNid == Integer.MAX_VALUE) {
114          if (Ts.get() != null) {
115             inferredNid = SnomedMetadataRf2.INFERRED_RELATIONSHIP_RF2.getLenient().getNid();
116             statedNid   = SnomedMetadataRf2.STATED_RELATIONSHIP_RF2.getLenient().getNid();
117          } else {
118 
119             // support for unit tests where Ts.get() may not be set.
120             inferredNid = 99;
121             statedNid   = 100;
122          }
123       }
124 
125       List<RelationshipIndexVersion> versions   = new ArrayList<>();
126       int                            end        = offset + data[offset + RECORD_LENGTH_OFFSET];
127       int                            firstStamp = offset + STAMP_INFERRED_HAS_GROUP_OFFSET;
128 
129       for (int i = firstStamp; i < end; i++) {
130          int stamp = data[i];
131 
132          stamp &= ~INFERRED_BITMASK;    // Clears the flag using bitwise AND and NOT
133          stamp &= ~GROUP_BITMASK;       // Clears the flag using bitwise AND and NOT
134 
135          int characteristicNid;
136          int group = 0;
137 
138          if ((data[i] & INFERRED_BITMASK) != 0) {
139             characteristicNid = inferredNid;
140          } else {
141             characteristicNid = statedNid;
142          }
143 
144          if ((data[i] & GROUP_BITMASK) != 0) {
145             group = data[i++];
146          }
147 
148          versions.add(new RelationshipIndexVersion(stamp, characteristicNid, group));
149       }
150 
151       return versions;
152    }

156          return true;
157       }
158 
159       return false;

162    boolean isActiveTaxonomyRelationship(ViewCoordinate vc, RelativePositionComputerBI computer)
163            throws IOException, ContradictionException {
164       if (vc.getIsaNid() == getTypeNid()) {
165          int           stampIndex = offset + DESTINATION_NID_OFFSET + 1;
166          int           recordEnd  = offset + data[offset + RECORD_LENGTH_OFFSET];
167          List<Integer> inferredStamps     = new ArrayList<>();
168          List<Integer> statedStamps     = new ArrayList<>();
169          List[] ArrayOfStampLists = new List[] {inferredStamps, statedStamps};
170 
171          while (stampIndex < recordEnd) {
172             if ((data[stampIndex] & GROUP_BITMASK) != 0) {
173 
174                // skip examination of this stamp, and skip the group.
175                // taxonomy rels are never grouped.
176                stampIndex++;
177             } else {
178                int stamp = data[stampIndex];
179 
180                stamp &= ~INFERRED_BITMASK;
181 
182                switch (vc.getRelationshipAssertionType()) {
183                case INFERRED :
184                   if ((data[stampIndex] & INFERRED_BITMASK) != 0) {
185                      inferredStamps.add(stamp);
186                   }
187 
188                   break;
189 
190                case STATED :
191                   if ((data[stampIndex] & INFERRED_BITMASK) == 0) {
192                      statedStamps.add(stamp);
193                   }
194 
195                   break;
196 
197                case INFERRED_THEN_STATED :
198                   if ((data[stampIndex] & INFERRED_BITMASK) != 0) {
199                      inferredStamps.add(stamp);
200                   }
201                   if ((data[stampIndex] & INFERRED_BITMASK) == 0) {
202                      statedStamps.add(stamp);
203                   }
204                   break;
205 
206                default :
207                   throw new UnsupportedOperationException();
208                }
209             }
210 
211             // next stampIndex
212             stampIndex++;
213          }
214 
215          // see if latest stamp is active;
216          for (List<Integer> stamps: ArrayOfStampLists) {
217               if (!stamps.isEmpty()) {
218                  VersionPoint latestStamp = null;
219 
220                  for (Integer stamp : stamps) {
221                     VersionPoint stampPoint = new VersionPoint(stamp);
222 
223                     if (computer.onRoute(stampPoint)) {
224                        if (latestStamp == null) {
225                           latestStamp = stampPoint;
226                        } else {
227                           switch (computer.relativePosition(stampPoint, latestStamp)) {
228                           case AFTER :
229                              latestStamp = stampPoint;
230 
231                              break;
232 
233                           case EQUAL:
234                           case BEFORE :
235                              break;
236 
237                           case CONTRADICTION :
238                              throw new ContradictionException("latestStamp: " + latestStamp + " stampPoint: "
239                                                               + stampPoint);
240 
241                           case UNREACHABLE :
242                              break;
243 
244                           default :
245                              throw new UnsupportedOperationException(computer.relativePosition(latestStamp,
246                                      latestStamp).toString());
247                           }
248                        }
249                     }
250                  }
251 
252                  if (latestStamp != null) {
253                     if (vc.getAllowedStatus().contains(latestStamp.getStatus())) {
254                        return true;
255                     }
256                  }
257               }
258           }
259       }
260 
261       return false;
262    }

266    public static int setGroupFlag(int stamp, boolean groupGtZero) {
267       if (groupGtZero) {
268          stamp |= GROUP_BITMASK;    // Sets the flag using bitwise OR
269       }
270 
271       return stamp;

274    public static int setInferredFlag(int stamp, boolean inferred) {
275       if (inferred) {
276          stamp |= INFERRED_BITMASK;    // Sets the flag using bitwise OR
277       }
278 
279       return stamp;

284    private class RelationshipIndexIterator implements Iterator<RelationshipIndexRecord> {
285       private boolean first = true;
286       private int     iteratorOffset;
287 
288       //~--- constructors -----------------------------------------------------
289 
290       public RelationshipIndexIterator(int iteratorOffset) {
291          this.iteratorOffset = iteratorOffset;
292       }
293 
294       //~--- methods ----------------------------------------------------------
295 
296       @Override
297       public RelationshipIndexRecord next() {
298          if (!first) {
299             iteratorOffset = iteratorOffset + data[iteratorOffset + RECORD_LENGTH_OFFSET];
300          }
301 
302          first = false;
303 
304          if (iteratorOffset < relationshipDataEnd) {
305             return new RelationshipIndexRecord(data, iteratorOffset, relationshipDataEnd);
306          }
307 
308          throw new NoSuchElementException();
309       }
310 
311       @Override
312       public void remove() {
313          throw new UnsupportedOperationException("Not supported yet.");
314       }
315 
316       //~--- get methods ------------------------------------------------------
317 
318       @Override
319       public boolean hasNext() {
320          if (first) {
321             if (iteratorOffset < relationshipDataEnd) {
322                return true;
323             }
324          }
325 
326          if (iteratorOffset + data[iteratorOffset + RECORD_LENGTH_OFFSET] < relationshipDataEnd) {
327             return true;
328          }
329 
330          return false;
331       }
