64      /**
65       * Applies a procedure to each key of the receiver, if any. Note: Iterates over the keys in no particular
66       * order. Subclasses can define a particular order, for example, "sorted by key". All methods which
67       * <i>can</i> be expressed in terms of this method (most methods can) <i>must guarantee</i> to use the
68       * <i>same</i> order defined by this method, even if it is no particular order. This is necessary so that,
69       * for example, methods <tt>keys</tt> and <tt>values</tt> will yield association pairs, not two
70       * uncorrelated lists.
71       *
72       * @param procedure the procedure to be applied. Stops iteration if the procedure returns <tt>false</tt>,
73       * otherwise continues.
74       * @return <tt>false</tt> if the procedure stopped before all keys where iterated over, <tt>true</tt>
75       * otherwise.
76       */
77      public abstract boolean forEachKey(UuidProcedure procedure);
78  
79      /**
80       * Applies a procedure to each (key,value) pair of the receiver, if any. Iteration order is guaranteed to
81       * be <i>identical</i> to the order used by method {@link #forEachKey(UuidProcedure)}.
82       *
83       * @param procedure the procedure to be applied. Stops iteration if the procedure returns <tt>false</tt>,
84       * otherwise continues.
85       * @return <tt>false</tt> if the procedure stopped before all keys where iterated over, <tt>true</tt>
86       * otherwise.
87       */
88      public boolean forEachPair(final UuidIntProcedure procedure) {
89          return forEachKey(new UuidProcedure() {
90              public boolean apply(long[] key) {
91                  return procedure.apply(key, get(key));
92              }
93          });
94      }

106     /**
107      * Returns the first key the given value is associated with. It is often a good idea to first check with
108      * {@link #containsValue(int)} whether there exists an association from a key to this value. Search order
109      * is guaranteed to be <i>identical</i> to the order used by method {@link #forEachKey(UuidProcedure)}.
110      *
111      * @param value the value to search for.
112      * @return the first key for which holds <tt>get(key) == value</tt>; returns <tt>Double.NaN</tt> if no
113      * such key exists.
114      */
115     public long[] keyOf(final int value) {
116         final long[] foundKey = new long[2];
117         boolean notFound = forEachPair(new UuidIntProcedure() {
118             public boolean apply(long[] iterKey, int iterValue) {
119                 boolean found = value == iterValue;
120                 if (found) {
121                     foundKey[0] = iterKey[0];
122                     foundKey[1] = iterKey[1];
123                 }
124                 return !found;
125             }
126 
127             @Override
128             public void close() {
129                 // nothing to do...
130             }
131         });
132         if (notFound) {
133             return null;
134         }
135         return foundKey;
136     }

172     /**
173      * Fills all pairs satisfying a given condition into the specified lists. Fills into the lists, starting
174      * at index 0. After this call returns the specified lists both have a new size, the number of pairs
175      * satisfying the condition. Iteration order is guaranteed to be <i>identical</i> to the order used by
176      * method {@link #forEachKey(UuidProcedure)}. <p> <b>Example:</b> <br>
177      *
178      * <pre>
179      * UuidIntProcedure condition = new UuidIntProcedure() { // match even values only
180      * 		public boolean apply(double key, int value) { return value%2==0; }
181      * 	}
182      * 	keys = (8,7,6), values = (1,2,2) --> keyList = (6,8), valueList = (2,1)</tt>
183      * </pre>
184      *
185      * @param condition the condition to be matched. Takes the current key as first and the current value as
186      * second argument.
187      * @param keyList the list to be filled with keys, can have any size.
188      * @param valueList the list to be filled with values, can have any size.
189      */
190     public void pairsMatching(final UuidIntProcedure condition,
191             final UuidArrayList keyList, final IntArrayList valueList) {
192         keyList.clear();
193         valueList.clear();
194 
195         forEachPair(new UuidIntProcedure() {
196             @Override
197             public boolean apply(long[] key, int value) {
198                 if (condition.apply(key, value)) {
199                     keyList.add(key);
200                     valueList.add(value);
201                 }
202                 return true;
203             }
204 
205             @Override
206             public void close() {
207                 // nothing to do...
208             }
209         });
210     }

212     /**
213      * Fills all keys and values <i>sorted ascending by key</i> into the specified lists. Fills into the
214      * lists, starting at index 0. After this call returns the specified lists both have a new size that
215      * equals <tt>this.size()</tt>. <p> <b>Example:</b> <br> <tt>keys = (8,7,6), values = (1,2,2) --> keyList
216      * = (6,7,8), valueList = (2,2,1)</tt>
217      *
218      * @param keyList the list to be filled with keys, can have any size.
219      * @param valueList the list to be filled with values, can have any size.
220      */
221     public void pairsSortedByKey(final UuidArrayList keyList,
222             final IntArrayList valueList) {
223         /*
224          * keys(keyList); values(valueList);
225          * 
226          * final double[] k = keyList.elements(); final int[] v =
227          * valueList.elements(); org.ihtsdo.Swapper swapper = new
228          * org.ihtsdo.Swapper() { public void swap(int a, int b) { int t1; double
229          * t2; t1 = v[a]; v[a] = v[b]; v[b] = t1; t2 = k[a]; k[a] = k[b]; k[b] =
230          * t2; } };
231          * 
232          * org.ihtsdo.function.IntComparator comp = new
233          * org.ihtsdo.function.IntComparator() { public int compare(int a, int b)
234          * { return k[a]<k[b] ? -1 : k[a]==k[b] ? 0 : 1; } };
235          * org.ihtsdo.MultiSorting.sort(0,keyList.size(),comp,swapper);
236          */
237 
238         // this variant may be quicker
239         // org.ihtsdo.map.OpenDoubleIntHashMap.hashCollisions = 0;
240         // System.out.println("collisions="+org.ihtsdo.map.OpenDoubleIntHashMap.hashCollisions);
241         keys(keyList);
242         keyList.sort();
243         valueList.setSize(keyList.size());
244         for (int i = keyList.size(); --i >= 0;) {
245             valueList.setQuick(i, get(keyList.getQuick(i)));
246         }
247         // System.out.println("collisions="+org.ihtsdo.map.OpenDoubleIntHashMap.hashCollisions);
248 
249     }

270     /**
271      * Returns a string representation of the receiver, containing the String representation of each key-value
272      * pair, sorted ascending by key.
273      */
274     @Override
275     public String toString() {
276         UuidArrayList theKeys = keys();
277         theKeys.sort();
278 
279         StringBuilder buf = new StringBuilder();
280         buf.append("[");
281         int maxIndex = theKeys.size() - 1;
282         for (int i = 0; i <= maxIndex; i++) {
283             long[] key = theKeys.get(i);
284             UUID uuidKey = new UUID(key[0], key[1]);
285             buf.append(uuidKey.toString());
286             buf.append("->");
287             buf.append(String.valueOf(get(key)));
288             if (i < maxIndex) {
289                 buf.append(", ");
290             }
291         }
292         buf.append("]");
293         return buf.toString();
294     }
