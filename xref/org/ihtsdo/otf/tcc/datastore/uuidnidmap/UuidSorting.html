33      /**
34       * Searches the list for the specified value using the binary search algorithm. The list must
35       * <strong>must</strong> be sorted (as by the sort method) prior to making this call. If it is not sorted,
36       * the results are undefined: in particular, the call may enter an infinite loop. If the list contains
37       * multiple elements equal to the specified key, there is no guarantee which of the multiple elements will
38       * be found.
39       *
40       * @param list the list to be searched.
41       * @param key the value to be searched for.
42       * @param from the leftmost search position, inclusive.
43       * @param to the rightmost search position, inclusive.
44       * @return index of the search key, if it is contained in the list; otherwise, <tt>(-(<i>insertion
45       * point</i>) - 1)</tt>. The <i>insertion point</i> is defined as the the point at which the value would
46       * be inserted into the list: the index of the first element greater than the key, or
47       * <tt>list.length</tt>, if all elements in the list are less than the specified key. Note that this
48       * guarantees that the return value will be &gt;= 0 if and only if the key is found.
49       * @see java.util.Arrays
50       */
51      public static int binarySearchFromTo(long[] list, long[] key, int from, int to) {
52          long[] midVal = new long[2];
53          while (from <= to) {
54              int mid = (from + to) / 2;
55              midVal[0] = list[mid * 2];
56              midVal[1] = list[mid * 2 + 1];
57              if (midVal[0] < key[0] || midVal[0] == key[0] && midVal[1] < key[1]) {
58                  from = mid + 1;
59              } else if (midVal[0] > key[0] || midVal[0] == key[0] && midVal[1] > key[1]) {
60                  to = mid - 1;
61              } else {
62                  return mid;
63              } // key found
64          }
65          return -(from + 1); // key not found.
66      }

68      /**
69       * Returns the index of the median of the three indexed chars.
70       */
71      private static int med3(long x[], int a, int b, int c, UuidComparatorBI comp) {
72          int ab = comp.compare(x[2 * a], x[2 * a + 1], x[2 * b], x[2 * b + 1]);
73          int ac = comp.compare(x[2 * a], x[2 * a + 1], x[2 * c], x[2 * c + 1]);
74          int bc = comp.compare(x[2 * b], x[2 * b + 1], x[2 * c], x[2 * c + 1]);
75          return (ab < 0 ? (bc < 0 ? b : ac < 0 ? c : a) : (bc > 0 ? b
76                  : ac > 0 ? c : a));
77      }

79      /**
80       * Sorts the specified range of the specified array of elements.
81       *
82       * <p> This sort is guaranteed to be <i>stable</i>: equal elements will not be reordered as a result of
83       * the sort. <p>
84       *
85       * The sorting algorithm is a modified mergesort (in which the merge is omitted if the highest element in
86       * the low sublist is less than the lowest element in the high sublist). This algorithm offers guaranteed
87       * n*log(n) performance, and can approach linear performance on nearly sorted lists.
88       *
89       * @param a the array to be sorted.
90       * @param fromIndex the index of the first element (inclusive) to be sorted.
91       * @param toIndex the index of the last element (exclusive) to be sorted.
92       * @throws IllegalArgumentException if <tt>fromIndex &gt; toIndex</tt>
93       * @throws ArrayIndexOutOfBoundsException if <tt>fromIndex &lt; 0</tt> or <tt>toIndex &gt; a.length</tt>
94       */
95      public static void mergeSort(long[] a, int fromIndex, int toIndex) {
96          rangeCheck(a.length, fromIndex, toIndex);
97          long aux[] = (long[]) a.clone();
98          mergeSort1(aux, a, fromIndex, toIndex);
99      }

104      * comparator (that is, <tt>c.compare(e1, e2)</tt> must not throw a <tt>ClassCastException</tt> for any
105      * elements <tt>e1</tt> and <tt>e2</tt> in the range). <p>
106      *
107      * This sort is guaranteed to be <i>stable</i>: equal elements will not be reordered as a result of the
108      * sort. <p>
109      *
110      * The sorting algorithm is a modified mergesort (in which the merge is omitted if the highest element in
111      * the low sublist is less than the lowest element in the high sublist). This algorithm offers guaranteed
112      * n*log(n) performance, and can approach linear performance on nearly sorted lists.
113      *
114      * @param a the array to be sorted.
115      * @param fromIndex the index of the first element (inclusive) to be sorted.
116      * @param toIndex the index of the last element (exclusive) to be sorted.
117      * @param c the comparator to determine the order of the array.
118      * @throws ClassCastException if the array contains elements that are not <i>mutually comparable</i> using
119      * the specified comparator.
120      * @throws IllegalArgumentException if <tt>fromIndex &gt; toIndex</tt>
121      * @throws ArrayIndexOutOfBoundsException if <tt>fromIndex &lt; 0</tt> or <tt>toIndex &gt; a.length</tt>
122      * @see Comparator

131     private static void mergeSort1(long src[], long dest[], int low, int high) {
132         int length = high - low;
133         // Insertion sort on smallest arrays
134         if (length < SMALL) {
135             for (int i = low; i < high; i++) {
136                 for (int j = i; j > low
137                         && (dest[2 * j - 2] > dest[2 * j]
138                         || (dest[2 * j - 2] == dest[2 * j] && dest[2 * j - 1] > dest[2 * j + 1])); j--) {
139                     swap(dest, j, j - 1);
140                 }
141             }
142             return;
143         }
144 
145         // Recursively sort halves of dest into src
146         int mid = (low + high) / 2;
147         mergeSort1(dest, src, low, mid);
148         mergeSort1(dest, src, mid, high);
149 
150         // If list is already sorted, just copy from src to dest. This is an
151         // optimization that results in faster sorts for nearly ordered lists.
152         if (src[2 * mid - 2] < src[2 * mid] || (src[2 * mid - 2] == src[2 * mid] && src[2 * mid - 1] <= src[2 * mid + 1])) {
153             System.arraycopy(src, 2 * low, dest, 2 * low, length * 2 + 1);
154             return;
155         }
156 
157         // Merge sorted halves (now in src) into dest
158         for (int i = low, p = low, q = mid; i < high; i++) {
159             if (q >= high || p < mid && (src[2 * p] < src[2 * q])
160                     || ((src[2 * p] == src[2 * q]) && (src[2 * p + 1] <= src[2 * q + 1]))) {
161                 dest[2 * i] = src[p * 2];
162                 dest[2 * i + 1] = src[p * 2 + 1];
163                 p++;
164             } else {
165                 dest[2 * i] = src[2 * q];
166                 dest[2 * i + 1] = src[2 * q + 1];
167                 q++;
168             }
169         }
170     }

172     private static void mergeSort1(long src[], long dest[], int low, int high,
173             UuidComparatorBI c) {
174         int length = high - low;
175 
176         // Insertion sort on smallest arrays
177         if (length < SMALL) {
178             for (int i = low; i < high; i++) {
179                 for (int j = i; j > low && c.compare(dest[2 * j - 2], dest[2 * j - 1], dest[j * 2], dest[j * 2 + 1]) > 0; j--) {
180                     swap(dest, j, j - 1);
181                 }
182             }
183             return;
184         }
185 
186         // Recursively sort halves of dest into src
187         int mid = (low + high) / 2;
188         mergeSort1(dest, src, low, mid, c);
189         mergeSort1(dest, src, mid, high, c);
190 
191         // If list is already sorted, just copy from src to dest. This is an
192         // optimization that results in faster sorts for nearly ordered lists.
193         if (c.compare(src[2 * mid - 2], src[2 * mid - 1], src[2 * mid], src[2 * mid + 1]) <= 0) {
194             System.arraycopy(src, low * 2, dest, low * 2, length * 2 + 1);
195             return;
196         }
197 
198         // Merge sorted halves (now in src) into dest
199         for (int i = low, p = low, q = mid; i < high; i++) {
200             if (q >= high || p < mid && c.compare(src[2 * p], src[2 * p + 1], src[2 * q], src[2 * q + 1]) <= 0) {
201                 dest[2 * i] = src[2 * p];
202                 dest[2 * i + 1] = src[2 * p + 1];
203                 p++;
204             } else {
205                 dest[2 * i] = src[2 * q];
206                 dest[2 * i + 1] = src[2 * q + 1];
207                 q++;
208             }
209         }
210     }

212     /**
213      * Sorts the specified range of the specified array of elements according to the order induced by the
214      * specified comparator. All elements in the range must be <i>mutually comparable</i> by the specified
215      * comparator (that is, <tt>c.compare(e1, e2)</tt> must not throw a <tt>ClassCastException</tt> for any
216      * elements <tt>e1</tt> and <tt>e2</tt> in the range). <p>
217      *
218      * The sorting algorithm is a tuned quicksort, adapted from Jon L. Bentley and M. Douglas McIlroy's
219      * "Engineering a Sort Function", Software-Practice and Experience, Vol. 23(11) P. 1249-1265 (November
220      * 1993). This algorithm offers n*log(n) performance on many data sets that cause other quicksorts to
221      * degrade to quadratic performance.
222      *
223      * @param a the array to be sorted.
224      * @param fromIndex the index of the first element (inclusive) to be sorted.
225      * @param toIndex the index of the last element (exclusive) to be sorted.
226      * @param c the comparator to determine the order of the array.
227      * @throws ClassCastException if the array contains elements that are not <i>mutually comparable</i> using
228      * the specified comparator.
229      * @throws IllegalArgumentException if <tt>fromIndex &gt; toIndex</tt>
230      * @throws ArrayIndexOutOfBoundsException if <tt>fromIndex &lt; 0</tt> or <tt>toIndex &gt; a.length</tt>
231      * @see Comparator
232      */
233     public static void quickSort(long[] a, int fromIndex, int toIndex,
234             UuidComparatorBI c) {
235         rangeCheck(a.length, fromIndex, toIndex);
236         quickSort1(a, fromIndex, toIndex - fromIndex, c);
237     }

239     /**
240      * Sorts the specified sub-array of chars into ascending order.
241      */
242     private static void quickSort1(long x[], int off, int len,
243             UuidComparatorBI comp) {
244         // Insertion sort on smallest arrays
245         if (len < SMALL) {
246             for (int i = off; i < len + off; i++) {
247                 for (int j = i; j > off && comp.compare(x[ 2 * j - 2], x[ 2 * j - 1], x[2 * j], x[2 * j + 1]) > 0; j--) {
248                     swap(x, j, j - 1);
249                 }
250             }
251             return;
252         }
253 
254         // Choose a partition element, v
255         int m = off + len / 2; // Small arrays, middle element
256         if (len > SMALL) {
257             int l = off;
258             int n = off + len - 1;
259             if (len > MEDIUM) { // Big arrays, pseudomedian of 9
260                 int s = len / 8;
261                 l = med3(x, l, l + s, l + 2 * s, comp);
262                 m = med3(x, m - s, m, m + s, comp);
263                 n = med3(x, n - 2 * s, n - s, n, comp);
264             }
265             m = med3(x, l, m, n, comp); // Mid-size, med of 3
266         }
267         long[] v = new long[2];
268         v[0] = x[m * 2];
269         v[1] = x[m * 2 + 1];
270 
271         // Establish Invariant: v* (<v)* (>v)* v*
272         int a = off, b = a, c = off + len - 1, d = c;
273         while (true) {
274             int comparison;
275             while (b <= c && (comparison = comp.compare(x[2 * b], x[2 * b + 1], v[0], v[1])) <= 0) {
276                 if (comparison == 0) {
277                     swap(x, a++, b);
278                 }
279                 b++;
280             }
281             while (c >= b && (comparison = comp.compare(x[2 * c], x[2 * c + 1], v[0], v[1])) >= 0) {
282                 if (comparison == 0) {
283                     swap(x, c, d--);
284                 }
285                 c--;
286             }
287             if (b > c) {
288                 break;
289             }
290             swap(x, b++, c--);
291         }
292 
293         // Swap partition elements back to middle
294         int s, n = off + len;
295         s = Math.min(a - off, b - a);
296         vecswap(x, off, b - s, s);
297         s = Math.min(d - c, n - d - 1);
298         vecswap(x, b, n - s, s);
299 
300         // Recursively sort non-partition-elements
301         if ((s = b - a) > 1) {
302             quickSort1(x, off, s, comp);
303         }
304         if ((s = d - c) > 1) {
305             quickSort1(x, n - s, s, comp);
306         }
307     }

309     /**
310      * Check that fromIndex and toIndex are in range, and throw an appropriate exception if they aren't.
311      */
312     private static void rangeCheck(int arrayLen, int fromIndex, int toIndex) {
313         if (fromIndex > toIndex) {
314             throw new IllegalArgumentException("fromIndex(" + fromIndex
315                     + ") > toIndex(" + toIndex + ")");
316         }
317         if (fromIndex < 0) {
318             throw new ArrayIndexOutOfBoundsException(fromIndex);
319         }
320         if (toIndex > arrayLen * 2 - 1) {
321             throw new ArrayIndexOutOfBoundsException(toIndex);
322         }
323     }

325     /**
326      * Swaps x[a] with x[b].
327      */
328     private static void swap(long x[], int a, int b) {
329         int aMsb = a * 2;
330         int aLsb = aMsb + 1;
331         int bMsb = b * 2;
332         int bLsb = bMsb + 1;
333         long[] t = new long[2];
334         t[0] = x[aMsb];
335         t[1] = x[aLsb];
336         x[aMsb] = x[bMsb];
337         x[aLsb] = x[bLsb];
338         x[bMsb] = t[0];
339         x[bLsb] = t[1];
340     }

342     /**
343      * Swaps x[a .. (a+n-1)] with x[b .. (b+n-1)].
344      */
345     private static void vecswap(long x[], int a, int b, int n) {
346         for (int i = 0; i < n; i++, a++, b++) {
347             swap(x, a, b);
348         }
