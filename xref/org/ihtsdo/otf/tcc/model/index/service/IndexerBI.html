35  /**
36   * The contract interface for indexing services.
37   * <br>
38   * <code>IndexerBI</code> implementations
39   * must not throw exceptions. Throwing exceptions could cause the underlying
40   * source data to corrupt. Since indexes can be regenerated, indexes should
41   * mark themselves as invalid somehow, and recreate themselves when necessary.
42   * @author aimeefurber
43   * @author kec
44   */
45  @Contract
46  public interface IndexerBI {
47      /**
48       *
49       * @param nid for the component that the caller wished to wait until it's
50       * document is added to the index.
51       * @return a <code>Callable&lt;Long&gt;</code> object that will block until this
52       * indexer has added the document to the index. The <code>call()</code> method
53       * on the object will return the index generation that contains the document,
54       * which can be used in search calls to make sure the generation is available
55       * to the searcher.
56       */
57      public IndexedGenerationCallable getIndexedGenerationCallable(int nid);
58  
59      /**
60       * Query index with no specified target generation of the index.
61       *
62       * @param query The query to apply.
63       * @param field The component field to be queried.
64       * @param sizeLimit The maximum size of the result list.
65       * @return a List of <code>SearchResult</codes> that contins the nid of the
66       * component that matched, and the score of that match relative to other
67       * matches.
68       * @throws IOException
69       * @throws ParseException
70       */
71      public List<SearchResult> query(String query, ComponentProperty field, int sizeLimit)
72              throws IOException, ParseException;
73  
74      /**
75       *
76       * @param query The query to apply.
77       * @param field The component field to be queried.
78       * @param sizeLimit The maximum size of the result list.
79       * @param targetGeneration target generation that must be included in the search
80       * or Long.MIN_VALUE if there is no need to wait for a target generation.
81       * @return a List of <code>SearchResult</codes> that contins the nid of the
82       * component that matched, and the score of that match relative to other
83       * matches.
84       * @throws IOException
85       * @throws ParseException
86       */
87      public List<SearchResult> query(String query, ComponentProperty field, int sizeLimit,
88              long targetGeneration)
89              throws IOException, ParseException;
90      
91      /**
92       *
93       * @return the name of this indexer.
94       */
95      public String getIndexerName();
96  
97      /**
98       * Checkpoints the index writer.
99       */
100     public void commitWriter();
101 
102     /**
103      * Close the index writer as part of normal shutdown.
104      */
105     public void closeWriter();
106 
107     /**
108      * Clear index, resulting in an empty index. Used prior to the
109      * environment recreating the index by iterating over all components
110      * and calling the <code>index(ComponentChronicleBI chronicle)</code>
111      * with each component of the iteration. May be used for initial index
112      * creation, or if indexing properties have changed.
113      */
114     public void clearIndex();
115 
116     /**
117      * Index the chronicle in a manner appropriate to the
118      * indexer implementation. The implementation is responsible to
119      * determine if the component is appropriate for indexing. All changed
120      * components will be sent to all indexers for indexing. The implementation
121      * must not perform lengthy operations on this thread.
122      *
123      * @return a <code>Future<Long></code>for the index generation to which this
124      * chronicle is attached.  If
125      * this chronicle is not indexed by this indexer, the Future returns
126      * <code>Long.MIN_VALUE<code>. The generation can be used with searchers
127      * to make sure that the component's indexing is complete prior to performing
128      * a search where the chronicle's results must be included.
129      */
130     public Future<Long> index(ComponentChronicleBI chronicle);
131     
132     /**
133      * Enables or disables an indexer. A disabled indexer will take
134      * no action when the index method is called. 
135      * @param enabled true if the indexer is enabled, otherwise false.
136      */
137     public void setEnabled(boolean enabled);
138     
139     /**
140      * 
141      * @return true if this indexer is enabled.
142      */
143     public boolean isEnabled();
144     
145 }
