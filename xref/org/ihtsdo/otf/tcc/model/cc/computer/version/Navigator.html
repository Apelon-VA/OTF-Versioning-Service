25      public <V extends ConceptComponent<?, ?>.Version> List<V> locateLatest(List<V> parts,
26              ViewCoordinate vc) throws IOException {
27          V latest = null;
28          OpenBitSet resultsPartSet = new OpenBitSet(parts.size());
29          
30              RelativePositionComputerBI mapper = RelativePositionComputer.getComputer(vc.getViewPosition());
31              OpenBitSet iteratorPartSet = new OpenBitSet(parts.size());
32              for (int i = 0; i < parts.size(); i++) {
33                  V part = parts.get(i);
34                  if (mapper.onRoute(part)) {
35                      if (latest == null) {
36                          latest = part;
37                          iteratorPartSet.set(i);
38                      } else {
39                          switch (mapper.relativePosition(latest, part)) {
40                              case BEFORE:
41                                  // nothing to do
42                                  break;
43  
44                              case CONTRADICTION:
45                                  iteratorPartSet.set(i);
46                                  break;
47  
48                              case AFTER:
49                                  latest = part;
50                                  iteratorPartSet.clear(0, Integer.MAX_VALUE);
51                                  iteratorPartSet.set(i);
52                                  break;
53  
54                              default:
55                                  break;
56                          }
57                      }
58                  }
59              }
60              resultsPartSet.or(iteratorPartSet);
61          
62          List<V> resultsList = new ArrayList<>((int) resultsPartSet.cardinality());
63          DocIdSetIterator resultsItr = resultsPartSet.iterator();
64          int id = resultsItr.nextDoc();
65          while (id != DocIdSetIterator.NO_MORE_DOCS) {
66              resultsList.add(parts.get(id));
67              id = resultsItr.nextDoc();
68          }
69          return resultsList;
70      }
