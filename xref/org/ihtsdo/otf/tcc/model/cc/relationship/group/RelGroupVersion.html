73     private void setupLatest() {
74        time = Long.MIN_VALUE;
75  
76        for (RelationshipVersionBI relV : getAllCurrentRelVersions()) {
77           if (relV.getTime() > time) {
78              time      = relV.getTime();
79              authorNid = relV.getAuthorNid();
80              pathNid   = relV.getPathNid();
81              status = relV.getStatus();
82              moduleNid = relV.getModuleNid();
83           }
84        }
85  
86        if (time == Long.MIN_VALUE) {
87           for (RelationshipChronicleBI rel : getRels()) {
88              for (RelationshipVersionBI relV : rel.getVersions(coordinate)) {
89                 if (relV.getTime() > time) {
90                    time      = relV.getTime();
91                    authorNid = relV.getAuthorNid();
92                    pathNid   = relV.getPathNid();
93                    status = relV.getStatus();
94                    moduleNid = relV.getModuleNid();
95                 }
96              }
97           }
98        }
