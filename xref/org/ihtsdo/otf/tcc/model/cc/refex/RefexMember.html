141         refexExtensionNid = input.readInt();
142         referencedComponentNid = input.readInt();
143         assert refexExtensionNid != Integer.MAX_VALUE;
144         assert referencedComponentNid != Integer.MAX_VALUE;
145         readMemberFields(input);
146 
147         int additionalVersionCount = input.readShort();
148 
149         if (additionalVersionCount > 0) {
150             if (revisions == null) {
151                 revisions = new RevisionSet<>(primordialStamp);
152             }
153 
154             for (int i = 0; i < additionalVersionCount; i++) {
155                 R r = readMemberRevision(input);

158                     revisions.add(r);
159                 }
160             }

241     @Override
242     public void writeToBdb(TupleOutput output, int maxReadOnlyStatusAtPositionNid) {
243         List<RefexRevision<R, C>> additionalVersionsToWrite = new ArrayList<>();
244 
245         if (revisions != null) {
246             for (RefexRevision<R, C> p : revisions) {
247                 if ((p.getStamp() > maxReadOnlyStatusAtPositionNid)
248                         && (p.getTime() != Long.MIN_VALUE)) {
249                     additionalVersionsToWrite.add(p);
250                 }
251             }
252         }
253 
254         assert refexExtensionNid != Integer.MAX_VALUE;
255         assert referencedComponentNid != Integer.MAX_VALUE;
256         output.writeInt(refexExtensionNid);
257         output.writeInt(referencedComponentNid);
258         writeMember(output);
259         output.writeShort(additionalVersionsToWrite.size());
260 
261         NidPairForRefex npr = NidPair.getRefexNidMemberNidPair(refexExtensionNid, nid);
262         try {
263             P.s.addXrefPair(referencedComponentNid, npr);
264         } catch (IOException ex) {
265            throw new RuntimeException(ex);
266         }
267 
268         for (RefexRevision<R, C> p : additionalVersionsToWrite) {
269             p.writeRevisionBdb(output);
270         }

307     public RefexMember<R, C>.Version getVersion(ViewCoordinate c) throws ContradictionException {
308         List<RefexMember<R, C>.Version> vForC = getVersions(c);
309 
310         if (vForC.isEmpty()) {
311             return null;
312         }
313 
314         if (vForC.size() > 1) {
315             vForC = c.getContradictionManager().resolveVersions(vForC);
316         }
317 
318         if (vForC.size() > 1) {
319             throw new ContradictionException(vForC.toString());
320         }
321 
322         if (!vForC.isEmpty()) {
323             return vForC.get(0);
324         }
325         return null;
326     }

337                 count = count + revisions.size();

340             ArrayList<Version> list = new ArrayList<>(count);

342             list.add(new Version(this));
343 
344             if (revisions != null) {
345                 for (RefexRevision rv : revisions) {
346                     list.add(new Version(rv));
347                 }
348             }
349 
350             versions = list;
351         }
352
