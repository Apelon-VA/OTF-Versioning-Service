147 
148         // nid, list size, and conceptNid are read already by the binder...
149         this.format = input.readString();
150 
151         int imageBytes = input.readInt();
152 
153         image = new byte[imageBytes];
154         input.read(image, 0, imageBytes);
155         textDescription = input.readString();

245     @Override
246     public void writeToBdb(TupleOutput output, int maxReadOnlyStatusAtPositionNid) {
247         List<MediaRevision> partsToWrite = new ArrayList<>();
248 
249         if (revisions != null) {
250             for (MediaRevision p : revisions) {
251                 if ((p.getStamp() > maxReadOnlyStatusAtPositionNid)
252                         && (p.getTime() != Long.MIN_VALUE)) {
253                     partsToWrite.add(p);
254                 }
255             }
256         }
257 
258         // Start writing
259         // conceptNid is the enclosing concept, does not need to be written.
260         output.writeString(format);
261         output.writeInt(image.length);
262         output.write(image);
263         output.writeString(textDescription);
264         output.writeInt(typeNid);
265         output.writeShort(partsToWrite.size());
266 
267         for (MediaRevision p : partsToWrite) {
268             p.writeRevisionBdb(output);
269         }
270     }

340         List<Media.Version> vForC = getVersions(c);
341 
342         if (vForC.isEmpty()) {

346         if (vForC.size() > 1) {

357     @Override
358     public List<Version> getVersions() {
359         if (versions == null) {
360             int count = 1;
361 
362             if (revisions != null) {
363                 count = count + revisions.size();
364             }
365 
366             ArrayList<Version> list = new ArrayList<>(count);
367 
368             if (getTime() != Long.MIN_VALUE) {
369                 list.add(new Version(this));
370             }
371 
372             if (revisions != null) {
373                 for (MediaRevision ir : revisions) {
374                     if (ir.getTime() != Long.MIN_VALUE) {
375                         list.add(new Version(ir));
376                     }
377                 }
378             }
379 
380             versions = list;
381         }
382 
383         return versions;
384     }
