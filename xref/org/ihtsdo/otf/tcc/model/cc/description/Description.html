161         String lastText = null;
162 
163         for (Description.Version desc : getVersions()) {
164             if (!desc.getText().equals(lastText)) {
165                 lastText = desc.getText();

182         text = input.readString();
183         typeNid = input.readInt();
184 
185         // nid, list size, and conceptNid are read already by the binder...
186         int additionalVersionCount = input.readShort();
187 
188         if (additionalVersionCount > 0) {
189             revisions = new RevisionSet<>(primordialStamp);

192                 DescriptionRevision dr = new DescriptionRevision(input, this);

195                     revisions.add(dr);

284     @Override
285     public void writeToBdb(TupleOutput output, int maxReadOnlyStatusAtPositionNid) {
286         List<DescriptionRevision> partsToWrite = new ArrayList<>();
287 
288         if (revisions != null) {
289             for (DescriptionRevision p : revisions) {
290                 if ((p.getStamp() > maxReadOnlyStatusAtPositionNid)
291                         && (p.getTime() != Long.MIN_VALUE)) {
292                     partsToWrite.add(p);
293                 }
294             }
295         }
296 
297         output.writeBoolean(initialCaseSignificant);
298         output.writeString(lang);
299         output.writeString(text);
300         output.writeInt(typeNid);
301         output.writeShort(partsToWrite.size());
302 
303         // conceptNid is the enclosing concept, does not need to be written.
304         for (DescriptionRevision p : partsToWrite) {
305             p.writeRevisionBdb(output);
306         }
307     }

357     @Override
358     public Description.Version getVersion(ViewCoordinate c) throws ContradictionException {
359         List<Description.Version> vForC = getVersions(c);
360 
361         if (vForC.isEmpty()) {
362             return null;
363         }
364 
365         if (vForC.size() > 1) {
366             vForC = c.getContradictionManager().resolveVersions(vForC);
367         }
368 
369         if (vForC.size() > 1) {
370             throw new ContradictionException(vForC.toString());
371         }
372 
373         if (!vForC.isEmpty()) {
374             return vForC.get(0);
375         }
376         return null;
377     }

379     @Override
380     public List<Version> getVersions() {
381         if (versions == null) {
382             int count = 1;
383 
384             if (revisions != null) {
385                 count = count + revisions.size();
386             }
387 
388             ArrayList<Version> list = new ArrayList<>(count);
389 
390             if (getTime() != Long.MIN_VALUE) {
391                 list.add(new Version(this));
392             }
393 
394             if (revisions != null) {
395                 for (DescriptionRevision rev : revisions) {
396                     if (rev.getTime() != Long.MIN_VALUE) {
397                         list.add(new Version(rev));
398                     }
399                 }
400             }
401 
402             versions = list;
403         }
404 
405         return versions;
406     }
